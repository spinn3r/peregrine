16  package peregrine.app.flow;
17  
18  import com.spinn3r.log5j.Logger;
19  import peregrine.Batch;
20  import peregrine.Job;
21  
22  /**
23   * <p>
24   * A 'flow' algorithm which starts off at a set of course nodes and finds
25   * connectivity between them.
26   *
27   * <p> The idea is that we start off at source nodes, find their outbound links,
28   * then merge them all together , then repeat.
29   *
30   * <p> One could almost certainly get similar output from pagerank with custom
31   * teleporation but this is quick and would be singificantly faster than using
32   * custom teleportation if we are able to significantly prune the size of the
33   * graph.
34   */
35  public class Flow extends Batch {
36      
37      private static final Logger log = Logger.getLogger();
38  
39      /**
40       */
41      private String graph = null;
42  
43      /**
44       * The number of iterations we should perform.
45       */
46      private int iterations = 5;
47  
48      public Flow( String input, String output, String sources, int iterations, boolean caseInsensitive ) {
49  
50          this.iterations = iterations;
51          
52          setName( Flow.class.getName() );
53  
54          String tmp = "/flow/connected.tmp";
55          
56          map( new Job().setDelegate( FlowInitJob.Map.class )
57                        .setInput( input )
58                        .setOutput( "shuffle:default" )
59                        .setParameters( "sources", sources,
60                                        "caseInsensitive", caseInsensitive )
61                        .setDescription( String.format( "Flow startup (sources=%s, caseInsensitive=%s)" , sources, caseInsensitive ) ) ) ;
62          
63          reduce( new Job().setDelegate( FlowInitJob.Reduce.class )
64                           .setCombiner( FlowInitJob.Reduce.class )
65                           .setInput( "shuffle:default" )
66                           .setOutput( tmp )
67                           .setDescription( String.format( "Writing output file %s", tmp ) ) );
68  
69          for( int i = 0; i < iterations; ++i ) {
70  
71              String desc = String.format( "Flow iter %s of %s" , i + 1, iterations );
72              
73              // now merge against the output and the input
74              merge( new Job().setDelegate( FlowIterJob.Merge.class )
75                              .setInput( tmp, input )
76                              .setOutput( "shuffle:default" )
77                              .setDescription( desc ) );
78  
79              reduce( new Job().setDelegate( FlowIterJob.Reduce.class )
80                               .setCombiner( FlowIterJob.Reduce.class )
81                               .setInput( "shuffle:default" )
82                               .setOutput( tmp )
83                               .setDescription( desc ) );
84              
85          }
86  
87          merge( new Job().setDelegate( FlowTermJob.Merge.class )
88                          .setInput( tmp, input )
89                          .setOutput( output )
90                          .setDescription( String.format( "Write final output %s", output ) ) );
91          
92          // now terminate the job by writing out the actual link data.
93          truncate( tmp ); 
94          
95      }
96  
97  }
