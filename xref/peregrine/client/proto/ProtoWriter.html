32  /**
33   * Client protocol response chunkWriter.  We use a regular chunk stream of key/value
34   * pairs as well as a trailer for meta data about the response.  There are just
35   * two blocks in this file format.  The data block which includes the key value
36   * pairs for the response and the meta block which includes tracing and scan
37   * metadata.
38   *
39   * <p>
40   * The ProtoWriter format does not have CRC32 as the frames/chunks are
41   * themselves checksummed.
42   * </p>
43   */
44  public class ProtoWriter extends BaseProto {
45  
46      public static ChannelBuffer MAGIC = ChannelBuffers.wrappedBuffer( "prp1".getBytes() );
47  
48      private Config config;
49  
50      private BufferedChannelBufferWritableWithChecksums channelBufferWritable;
51  
52      private DataSection dataSection;
53  
54      private MetaSection metaSection;
55  
56      public ProtoWriter(Config config, ChannelBufferWritable target, boolean checksum ) throws IOException {
57  
58          this.config = config;
59          this.channelBufferWritable
60                  = new BufferedChannelBufferWritableWithChecksums( target,
61                                                                    DefaultChunkWriter.BUFFER_SIZE,
62                                                                    checksum );
63  
64          channelBufferWritable.setCloseable(false);
65          channelBufferWritable.write( MAGIC.duplicate() );
66  
67          dataSectionOffset = (int)channelBufferWritable.length();
68  
69      }
70  
71      public DataSection createDataSection() throws IOException {
72          dataSection = new DataSectionImpl();
73          return dataSection;
74      }
75  
76      /**
77       * Create a section for metadata values.  Once the MetaSection is created the
78       * DataSection is invalid and can no longer be written.
79       *
80       * <p>
81       * Metadata should not be binary and should be printable.  Binary metadata is
82       * possible and the client should be able to handle this situation by Hex
83       * encoding the result.
84       * </p>
85       *
86       */
87      public MetaSection createMetaSection() throws IOException {
88  
89          dataSection.close();
90  
91          dataSectionLength = (int)(channelBufferWritable.length()) - dataSectionOffset;
92  
93          metaSectionOffset = (int)channelBufferWritable.length();
94  
95          metaSection = new MetaSectionImpl();
96          return metaSection;
97  
98      }
99  
100     public void close() throws IOException {
101 
102         metaSection.close();
103         metaSectionLength = (int)(channelBufferWritable.length()) - metaSectionOffset;
104 
105         channelBufferWritable.flush();
106 
107         channelBufferWritable.write( wrap( dataSectionOffset ) );
108         channelBufferWritable.write( wrap( dataSectionLength ) );
109         channelBufferWritable.write( wrap( metaSectionOffset ) );
110         channelBufferWritable.write( wrap( metaSectionLength ) );
111 
112         channelBufferWritable.setCloseable(true);
113         channelBufferWritable.close();
114 
115     }
116 
117     public long length() {
118         return channelBufferWritable.length();
119     }
120 
121     class Section extends DefaultChunkWriter {
122 
123         Section() throws IOException {
124             super( config, channelBufferWritable );
125         }
126 
127     }
128 
129     // mostly a tag interface to avoid exposing all of DefaultChunkWriter.
130     interface DataSection extends SequenceWriter {
131     }
132 
133     class DataSectionImpl extends Section implements DataSection {
134 
135         public DataSectionImpl() throws IOException {
136             super();
137         }
138 
139         @Override
140         public void write( StructReader key, StructReader value ) throws IOException {
141 
142             if ( metaSectionOffset != -1 )
143                 throw new IOException( "Has meta section." );
144 
145             super.write( key, value );
146 
147         }
148 
149     }
150 
151     // mostly a tag interface to avoid exposing all of DefaultChunkWriter.
152     interface MetaSection extends SequenceWriter {
153     }
154 
155     class MetaSectionImpl extends Section implements MetaSection {
156 
157         public MetaSectionImpl() throws IOException {
158             super();
159         }
160 
161         @Override
162         public void write( StructReader key, StructReader value ) throws IOException {
163 
164             super.write( key, value );
165 
166         }
167 
168     }
