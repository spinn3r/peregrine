16  package peregrine.config;
17  
18  import com.spinn3r.log5j.Logger;
19  
20  import java.util.*;
21  
22  /**
23   * Manages layout for the partition system by distributing partitions across host.
24   */
25  public class PartitionLayoutEngine {
26  
27      private static final Logger log = Logger.getLogger();
28  
29      /**
30       * The host to partition matrix.
31       */
32      Map<Host,List<Partition>> partitionsByHost = new HashMap();
33  
34      Map<Partition,List<Host>> hostsByPartition = new HashMap();
35  
36      protected Map<Host,List<Replica>> replicasByHost = new HashMap();
37      
38      protected Map<Partition,List<Replica>> replicasByPartition = new HashMap();
39  
40      protected int nr_hosts;
41      protected int nr_partitions_per_host;
42      protected int nr_replicas;
43  
44      /**
45       * The total number of hosts that can fail before we are below our minimum
46       * replica count.
47       */
48      protected int spare_hosts;
49      
50      protected List<Host> hosts;
51  
52      protected Config config;
53      
54      public PartitionLayoutEngine( Config config ) {
55  
56          // build the hosts index
57          this.hosts = new ArrayList();
58          this.hosts.addAll( config.getHosts() );
59  
60          // sort the hosts. In the Set they are not sorted.  The implementation
61          // of this was initially a HashSet which may change but we should always
62          // sort the hosts to make testing easier since we know how the partition
63          // layout apriori.
64          
65          Collections.sort( this.hosts );
66          
67          this.nr_hosts                = hosts.size();
68          this.nr_partitions_per_host  = config.getPartitionsPerHost();
69          this.nr_replicas             = config.getReplicas();
70          this.config                  = config;
71  
72          if ( nr_hosts == 0 )
73              throw new PartitionLayoutException( "No hosts defined." );
74  
75          if ( nr_replicas == 0 || nr_replicas > 3 )
76              throw new PartitionLayoutException( "Invalid nr_replicas defined: " + nr_replicas );
77  
78      }
79  
80      public void build() {
81  
82          log.info( "Building partition layout with %,d partitions_per_host and %s replicas." , nr_partitions_per_host, nr_replicas );
83  
84          // I think the last partition will have (nr_hosts * nr_partitions_per_host) %
85          // nr_replicas copies and we can just evenly hand these out to
86          // additional hosts
87          
88          if ( nr_hosts < nr_replicas ) {
89              throw new PartitionLayoutException( String.format( "Incorrect number of hosts (nr_hosts < nr_replicas).  (nr_hosts=%s, nr_replicas=%s)",
90                                                                 nr_hosts,
91                                                                 nr_replicas ) );
92          }
93  
94          if ( nr_partitions_per_host % nr_replicas != 0 ) {
95              throw new PartitionLayoutException( "nr_partitions_per_host % nr_replicas must equal zero to fit partitions correctly." );
96          }
97  
98          int min_hosts = nr_replicas * nr_partitions_per_host;
99          
100         if ( nr_hosts <= min_hosts ) {
101             log.warn( "For maximum parallel recovery, your nr_hosts should be > nr_replicas * nr_partitions_per_host" );
102         }
103 
104         spare_hosts = nr_hosts - nr_partitions_per_host;
105 
106         //ideal_hosts = nr_hosts - ((nr_partitions_per_host * 2) + 1);
107         
108         // On startup... We need to print the number of machines we can handle
109         // failing without falling below the minimum number of replicas...
110 
111         if ( spare_hosts <= 0 && nr_hosts != 1 ) {
112             throw new PartitionLayoutException( getSpareHostsMsg() );
113         } else {
114             log.info( "%s", getSpareHostsMsg() );
115         }
116 
117         //if ( ideal_hosts 
118         
119         int nr_primary_per_host = nr_partitions_per_host / nr_replicas;
120 
121         log.info( "nr_partitions_per_host: %,d", nr_partitions_per_host );
122         
123         // startup the partitionsByHost.
124         for( Host host : hosts ) {
125             partitionsByHost.put( host, new ArrayList() ); 
126             replicasByHost.put( host, new ArrayList() ); 
127         }
128 
129         // startup the hostsByPartition lookup...
130         int nr_partitions = nr_primary_per_host * nr_hosts;
131 
132         log.info( "Number of unique partitions: %,d", nr_partitions );
133         
134         for( int i = 0; i < nr_partitions; ++i ) {
135             hostsByPartition.put( new Partition( i ), new ArrayList() ); 
136             replicasByPartition.put( new Partition( i ), new ArrayList() ); 
137         }
138 
139         // now create the replicas... 
140         for( int i = 0; i < nr_replicas; ++i ) {
141 
142             if ( i % 2 == 0 ) {
143 
144                 int host_offset = (i * nr_primary_per_host) + i;
145 
146                 if ( host_offset != 0 )
147                     --host_offset;
148 
149                 for( int j = 0; j < nr_partitions; ++j ) {
150 
151                     int host_id = (j % nr_hosts) + host_offset;
152 
153                     if ( host_id >= nr_hosts )
154                         host_id = host_id - nr_hosts;
155                     
156                     Host host = hosts.get( host_id );
157                     
158                     associate( host, new Partition( j ), i );
159 
160                 }
161 
162             } else { 
163             
164                 // we don't start at the first host to avoid placing
165                 // the same partitions that it currently hosts on it.
166 
167                 int nr_columns = nr_primary_per_host;
168                 
169                 for( int j = 0; j < nr_columns; ++j ) {
170 
171                     int host_id = j + 1;
172 
173                     for( int k = 0; k < nr_hosts ; ++k ) {
174 
175                         Host host = hosts.get( host_id );
176 
177                         associate( host, new Partition( (j * nr_hosts) + k ), i );
178 
179                         host_id = (host_id == nr_hosts - 1) ? 0 : host_id + 1;
180                         
181                     }
182                         
183                 }
184 
185             }
186                 
187         }
188 
189         assertCorrectLayout();
190         
191     }
192 
193     protected String getSpareHostsMsg() {
194 
195         return String.format( "%,d hosts can fail before you risk partition lost due to nr_replicas." ,
196                               spare_hosts );
197 
198     }
199 
200     /**
201      * Assert that we're running with the correct partition layout.  If not then
202      * we need to throw an exception as we can't run with an incorrect config.
203      * Since these are TWO different algorithms the chances of catching bugs is
204      * much higher.
205      */
206     private void assertCorrectLayout() {
207 
208         for( Host host: replicasByHost.keySet() ) {
209 
210             List<Partition> partitions = partitionsByHost.get( host );
211 
212             Set<Host> hosts = new HashSet();
213 
214             for( Partition partition : partitions ) {
215 
216                 List<Replica> replicas = replicasByPartition.get( partition );
217 
218                 for( Replica replica : replicas ) {
219 
220                     // replicas for this host don't count.
221                     if( replica.getHost().equals( host ) )
222                         continue;
223 
224                     hosts.add( replica.getHost() );
225 
226                 }
227                 
228             }
229 
230             if ( nr_replicas > 1 && hosts.size() < nr_partitions_per_host ) {
231                 throw new PartitionLayoutException( String.format( "Replica config for %s is too small: %s", host, hosts ) );
232             }
233             
234         }
235         
236     }
237     
238     /**
239      * Associate a given partition with a given host.  This will perform the
240      * hostsByPartition and reverse lookups.
241      */
242     private void associate( Host host, Partition partition, int priority ) {
243         
244         partitionsByHost.get( host ).add( partition );
245         hostsByPartition.get( partition ).add( host );
246 
247         // now create a Replica for this partition.
248 
249         Replica replica = new Replica( host, partition, priority );
250         
251         replicasByHost.get( host ).add( replica );
252         replicasByPartition.get( partition ).add( replica );
253         
254     }
255 
256     public Membership toMembership() {
257 
258         return new Membership( config, this, hostsByPartition, partitionsByHost, replicasByHost );
259 
260     }
