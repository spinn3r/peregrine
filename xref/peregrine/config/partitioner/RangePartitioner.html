16  package peregrine.config.partitioner;
17  
18  import peregrine.StructReader;
19  import peregrine.config.Config;
20  import peregrine.config.Partition;
21  
22  /**
23   * Partitions by range based on the key and the partition distribution.
24   */
25  public class RangePartitioner extends BasePartitioner {
26  
27      private static final int WIDTH = (int)Math.pow( 2, 16 );
28      
29  	private double range;
30  	
31  	@Override
32      public void init( Config config ) {
33  
34  		super.init( config );
35  
36          //TODO: migrate this to BigInteger so that I can support weights in
37          //partitions so that we can easily split ranges.  We can then use
38          //TreeMap.ceilingEntry to route within the partition space.
39          
40      	this.range = WIDTH / (double)nr_partitions;	
41  
42      }
43  
44  	@Override
45  	public Partition partition( StructReader key, StructReader value ) {
46  		
47  		byte[] bytes = key.toByteArray();
48  		
49  		// the domain of the route function...  basically the key space as an
50  		// integer so that we can place partitions within that space.
51  
52          int val = ((((int) bytes[7]) & 0xFF) << 0) +
53                    ((((int) bytes[6]) & 0xFF) << 8)
54              ;
55  
56  		int part = (int)(val / range);
57  
58          //TODO: technically we should require this but it will slow down mapping.
59          //if ( part > nr_partitions )
60          //    throw new RuntimeException( String.format( "Unable to correctly route to partition. %s vs %s",
61          //                                               part, nr_partitions ) );
62          
63  		return new Partition( part );
64  		
65  	}
66  
67  }
