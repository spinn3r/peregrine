16  package peregrine.console.fs;
17  
18  import peregrine.StructReader;
19  import peregrine.StructReaders;
20  import peregrine.io.SequenceReader;
21  import peregrine.io.chunk.DefaultChunkReader;
22  import peregrine.io.util.Closer;
23  import peregrine.util.Base64;
24  import peregrine.util.Getopt;
25  
26  import java.io.File;
27  import java.util.List;
28  
29  /**
30   * Cat a file on the filesytem.  This reads from a local PFS directory file
31   * (someday this may be globally via HTTP across the entire filesystem) and cats
32   * the file to standard out.
33   * 
34   * <p>
35   * 
36   * This can be used for debug purposes so that one can understand what is stored
37   * on the filesystem without having to write additional map reduce jobs.
38   */
39  public class Cat {
40  
41      // FIXME: this should completely go away in favor of scan.
42      
43      public static void help() {
44  
45          System.out.printf( "Usage: fscat [OPTION] [FILE]\n" );
46          System.err.printf( "Simple Peregrine filesystem 'cat' command \n" );
47          System.err.printf( "\n" );
48          System.out.printf( "  --render   Render the value(s) in the following format.\n" );
49          System.out.printf( "             Multiple columns are separated by a comma (,)\n" );
50          System.err.printf( "\n" );
51          System.out.printf( "  --limit    Limit results to the first N entries.\n" );        
52          System.out.printf( "\n" );
53          System.out.printf( "The --render option supports a comma separated list\n" );
54          System.out.printf( "of values to print from the byte stream.\n" );
55          System.out.printf( "\n" );
56          
57      }
58      
59      public static void main( String[] args ) throws Exception {
60  
61          Getopt getopt = new Getopt( args );
62  
63          if ( getopt.getBoolean( "help" ) ) {
64              help();
65              System.exit( 1 );
66          }
67          
68          String render  = getopt.getString( "render" , "base64" );
69          int limit      = getopt.getInt( "limit", -1 );
70  
71          //TODO: go over ALL paths specified so the user can cat multiple files.
72          String path = getopt.getValues().get( 0 );
73  
74          SequenceReader reader = null;
75  
76          try {
77              
78              reader = new DefaultChunkReader( new File( path ) );
79  
80              int id = 0;
81              while ( reader.hasNext() ) {
82                  
83                  reader.next();
84                  
85                  StructReader key   = reader.key();
86                  StructReader value = reader.value();
87                  
88                  System.out.printf( "%s = %s\n", Base64.encode( key.toByteArray() ), format( render, value ) );
89  
90                  if ( limit != -1 && ++id >= limit )
91                      break;
92                  
93              }
94  
95          } finally {
96              new Closer( reader ).close();
97          }
98  
99      }
100 
101     /*
102 
103        TODO:
104 
105        - ability to name columns so that the result shows up correctly in the
106          output with named columns.
107 
108        - printf format specifiers %02d would be nice.  Maybe just use printf
109          formatters directly.
110 
111        - byte[] needs encoding mechanisms.
112 
113        - perhaps store the record format WITH the files so that we never have
114          the type system when cating a file.
115 
116        - Maybe split them on format specifiers.  For example %d,%d,%2.2f 
117          
118      */
119     private static String format( String render, StructReader value ) {
120 
121         StringBuilder buff = new StringBuilder();
122 
123         /*
124          * byte
125          * short
126          * varint
127          * int
128          * long
129          * float
130          * double
131          * boolean
132          * char
133          * string
134          * hashcode
135          */
136              
137         for ( String r : render.split( "," ) ) {
138 
139             if ( buff.length() > 0 )
140                 buff.append( " " );
141 
142             if ( r.matches( "(?i)%.*d" ) ) {
143                 buff.append( String.format( r, value.readInt() ) );
144                 continue;
145             }
146 
147             if ( r.matches( "(?i)%.*f" ) ) {
148                 buff.append( String.format( r, value.readDouble() ) );
149                 continue;
150             }
151 
152             if ( r.matches( "(?i)%.*s" ) ) {
153                 buff.append( String.format( r, value.readString() ) );
154                 continue;
155             }
156 
157             //TODO we could use reflection here.  Java really needs syntactic
158             //sugar for reflection.
159             
160             if ( "byte".equals( r ) ) {
161                 buff.append( value.readByte() );
162                 continue;
163             }
164 
165             if ( "short".equals( r ) ) {
166                 buff.append( value.readShort() );
167                 continue;
168             }
169 
170             if ( "varint".equals( r ) ) {
171                 buff.append( value.readVarint() );
172                 continue;
173             }
174 
175             if ( "long".equals( r ) ) {
176                 buff.append( value.readLong() );
177                 continue;
178             }
179 
180             if ( "float".equals( r ) ) {
181                 buff.append( value.readFloat() );
182                 continue;
183             }
184 
185             if ( "double".equals( r ) ) {
186                 buff.append( String.format( "%f", value.readDouble() ) );
187                 continue;
188             }
189 
190             if ( "boolean".equals( r ) ) {
191                 buff.append( value.readBoolean() );
192                 continue;
193             }
194 
195             if ( "char".equals( r ) ) {
196                 buff.append( value.readChar() );
197                 continue;
198             }
199 
200             if ( "string".equals( r ) ) {
201                 buff.append( value.readString() );
202                 continue;
203             }
204 
205             if ( "int".equals( r ) ) {
206                 buff.append( value.readInt() );
207                 continue;
208             }
209 
210             if ( "byte".equals( r ) ) {
211                 buff.append( value.readByte() );
212                 continue;
213             }
214 
215             if ( "hashcode".equals( r ) ) {
216                 buff.append( Base64.encode( value.readHashcode() ) );
217                 continue;
218             }
219 
220             if ( "wrapped:string".equals( r ) ) {
221 
222                 List<StructReader> unwrapped = StructReaders.unsplice( value );
223 
224                 for( StructReader u : unwrapped ) {
225                     buff.append( u.readString() + " " );
226                 }
227 
228                 continue;
229             }
230 
231             if ( "wrapped:hashcode".equals( r ) ) {
232 
233                 List<StructReader> unwrapped = StructReaders.unsplice( value );
234 
235                 for( StructReader u : unwrapped ) {
236                     buff.append( Base64.encode( u.readHashcode() ) + " " );
237                 }
238 
239                 continue;
240             }
241 
242             buff.append( String.format( "%s:", value.length() ) );
243             buff.append( Base64.encode( value.toByteArray() ) );
244 
245         }
246 
247         return buff.toString();
248         
249     }
