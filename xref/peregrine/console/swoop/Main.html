65      public static String progress( double perc, int width ) {
66  
67          StringBuilder buff = new StringBuilder();
68  
69          buff.append( "[" );
70          
71          int cutoff = (int)(width * ( perc / (double)100 ));
72          
73          for( int i = 0; i < width; ++i ) {
74  
75              if ( i < cutoff ) {
76                  buff.append( "*" );
77              } else {
78                  buff.append( " " );
79              }
80              
81          }
82  
83          buff.append( "]" );
84  
85          buff.append( String.format( " %4.1f%%", perc ) );
86          
87          return buff.toString();
88          
89      }

159     private String getFirstLine( String data, String _default ) {
160 
161         if ( Strings.empty( data ) )
162             return _default;
163 
164         data = data.trim();
165         
166         String[] split = data.trim().split( "\n" );
167 
168         if ( split.length == 0 )
169             return _default;
170         
171         return split[0];
172         
173     }

175     private String getDuration( long started, long duration ) {
176 
177         if ( duration == 0 && started != 0 )
178             duration = System.currentTimeMillis() - started;
179         
180         if ( duration <= 0 )
181             return "";
182 
183         return new Duration( duration ).toString();
184

258     public void formatBatch( Batch batch, boolean extended, Formatter fmt ) {
259 
260         int nr_jobs = batch.getJobs().size();
261         int nr_complete = 0;
262         
263         for( Job job : batch.getJobs() ) {
264             
265             if ( job.getState().equals( JobState.COMPLETED ) )
266                 ++nr_complete;
267             
268         }
269 
270         double perc_complete = 100 * (nr_complete / (double)nr_jobs);
271 
272         fmt.printf( "Currently executing batch:" );
273         
274         fmt.newline();
275 
276         formatBatchOverviewHeaders( fmt );
277         formatBatchOverview( batch, fmt );
278 
279         fmt.newline();
280 
281         fmt.printf( 4, "nr complete jobs:     %,d" , nr_complete );
282         fmt.printf( 4, "batch perc complete:  %s" ,  progress( perc_complete, PROGRESS_WIDTH ) );
283 
284         //add job perc complete which comes from the scheduler.
285 
286         if ( response.getSchedulerStatusResponse() != null ) {
287             fmt.printf( 4, "job perc complete:    %s" , progress( response.getSchedulerStatusResponse().getProgress(), PROGRESS_WIDTH ) );
288         }
289 
290         fmt.newline();
291 
292         formatJobsHeaders( fmt );
293 
294         for ( Job job : batch.getJobs() ) {
295 
296             formatJob( job, fmt );
297 
298             if ( extended ) {
299                 formatJobExtended( job, fmt );
300             } 
301 
302         }
303 
304         fmt.newline();
305 
306         Job job = batch.getExecutingJob();
307 
308         if ( job != null && extended == false ) {
309 
310             //TODO cause, comparator, maxChunks, partitioner, combiner, 
311             fmt.printf( "Currently executing job '%s':", job.getName() );
312 
313             fmt.newline();
314             
315             formatJobExtended( job, fmt );
316 
317         }
318

361     /**
362      * Read the status messgae from the controller and update global variables.
363      */
364     private void readStatusFromController() {
365 
366         cause = null;
367         response = new ControllerStatusResponse();
368 
369         try {
370         
371             RpcCient rpcCient = new RpcCient( config );
372             
373             Message message = new Message();
374             message.put( "action", "status" );
375             
376             Message result = rpcCient.invoke( config.getController(), "controller", message );
377             
378             response.fromMessage( result );
379 
380         } catch ( Exception e ) {
381             cause = e;
382         }
383 
384     }

386     private Batch findBatch( long identifier ) {
387 
388         List<Batch> list = new ArrayList();
389         list.addAll( response.getHistory() );
390         list.addAll( response.getPending() );
391 
392         if ( response.getExecuting() != null )
393             list.add( response.getExecuting() );
394         
395         for( Batch batch : list ) {
396 
397             if ( batch.getIdentifier() == identifier )
398                 return batch;
399             
400         }
401 
402         return null;
403         
404     }

457     public void exec() {
458 
459         while( true ) {
460 
461             if ( quit )
462                 break;
463             
464             curses.clear();
465 
466             readStatusFromController();
467             
468             //TODO: include scheduler cluster state.
469 
470             try {
471 
472                 String state = "ONLINE";
473 
474                 if ( cause != null )
475                     state = "OFFLINE: " + cause.getMessage();
476 
477                 curses.mvaddstr( y_pos++, 0, String.format( "%s", "Controller:" ) );
478                 curses.mvaddstr( y_pos++, 4, String.format( "%-20s %s", "Last refresh:", new Date() ) );
479                 curses.mvaddstr( y_pos++, 4, String.format( "%-20s %s", "Host:", config.getController() ) );
480                 curses.mvaddstr( y_pos++, 4, String.format( "%-20s %s", "State:", state ) );
481 
482                 if ( cause != null )
483                     continue;
484 
485                 long now = System.currentTimeMillis();
486 
487                 curses.mvaddstr( y_pos++, 4, String.format( "%-20s %s", "Uptime:", getDuration( now, now - response.getStarted() ) ) );
488                 curses.mvaddstr( y_pos++, 4, String.format( "%-20s %s", "Historical batches:", response.getHistory().size() ) );
489                 curses.mvaddstr( y_pos++, 4, String.format( "%-20s %s", "Pending batches:", response.getPending().size() ) );
490 
491                 ++y_pos;
492                 
493                 // TODO compute some stats on the history ... # failed, #
494                 // completed.  Failure rate.
495 
496                 // TODO: the table should have 3 options:
497                 //
498                 //  - (E) show what is currently executing 
499                 //  - (P) show pending batches
500                 //  - (H) show what is in history
501                 //
502                 //  - (X) they should ALL have the option to show extended metadata 
503 
504                 switch( mode ) {
505                     case AUTO:
506                         doAuto();
507                         break;
508                     case HISTORY:
509                         doHistory();
510                         break;
511                     case PENDING:
512                         doPending();
513                         break;
514                     case EXECUTING:
515                         doExecuting();
516                         break;
517                     case LAST:
518                         doLast();
519                         break;
520                 }
521 
522             } finally {
523 
524                 // write one more line so the cursur is in a sane place.
525                 ++y_pos;
526                 curses.mvaddstr( y_pos++, 0, getModeline() );
527                 curses.mvaddstr( y_pos++, 0, "" );
528                 curses.refresh();
529 
530                 y_pos = 0;
531 
532                 char read = (char)curses.getch();
533 
534                 if ( read > -1 )
535                     read = Character.toLowerCase( read );
536                 
537                 switch( read ) {
538 
539                     case 'a':
540                         mode = Mode.AUTO;
541                         break;
542 
543                     case 'x':
544                         mode = Mode.EXECUTING;
545                         break;
546 
547                     case 'h':
548                         mode = Mode.HISTORY;
549                         break;
550 
551                     case 'p':
552                         mode = Mode.PENDING;
553                         break;
554 
555                     case 'l':
556                         mode = Mode.LAST;
557                         break;
558 
559                     case 'q':
560                         quit();
561                         System.exit( 0 );
562                         break; /* not needed */
563                         
564                     default:
565                         break;
566                         
567                 }
568 
569             }
570             
571         }
572 
573     }

575     public static void main( String[] args ) throws Exception {
576 
577         // TODO:
578         //
579         //  q      should quit
580 
581         new Initializer().logger( "swoop" );
582 
583         Getopt getopt = new Getopt( args );
584 
585         Main swoop = new Main();
586         swoop.config = ConfigParser.parse( args );
587 
588         if ( getopt.containsKey( "batch" ) ) {
589 
590             swoop.readStatusFromController();
591             long identifier = getopt.getLong( "batch" );
592 
593             Batch batch = swoop.findBatch( identifier );
594 
595             if ( batch != null ) {
596 
597                 Formatter fmt = new Formatter();
598                 swoop.formatBatch( batch , true, fmt );
599                 System.out.printf( "%s\n", fmt.toString() );
600                 
601             } else {
602                 System.out.printf( "Batch %s not found.\n", identifier );
603             }
604 
605             return;
606             
607         }
608 
609         curses.init();
610 
611         Runtime.getRuntime().addShutdownHook( new Thread() {
612 
613                 public void run() {
614                     quit();
615                 }
616                 
617             } );
618 
619         swoop.exec();
620 
621     }
