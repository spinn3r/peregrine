16  package peregrine.controller;
17  
18  import com.spinn3r.log5j.Logger;
19  import peregrine.config.Config;
20  import peregrine.config.Host;
21  import peregrine.util.MarkMap;
22  import peregrine.util.MarkSet;
23  
24  /**
25   * Keeps track of system failures in the controller.  If a majority of machines 
26   * have sent gossip that this machine has failed then mark it offline and 
27   * no longer schedule work to it or attempt writes.
28   * 
29   * @author burton
30   *
31   */
32  public class Gossip extends MarkMap<Host,MarkSet<Host>> {
33  
34  	private static final Logger log = Logger.getLogger();
35  
36      public static double QUORUM = 0.5;
37      
38  	private Config config;
39  	
40  	private Online online;
41  	
42  	private Offline offline;
43  		
44  	public Gossip( Config config, Online online, Offline offline ) {		
45  		
46  		this.config = config;
47  		this.online = online;
48  		this.offline = offline;
49  				
50  		for( Host host : config.getHosts() ) {
51  			map.put( host, new MarkSet() );
52  		}
53  		
54  	}
55  
56  	public void mark( Host reporter, Host failed ) {
57  		
58  		// mark the machine as failed... if a majority of host have marked this
59  		// machine as being failed, take it out of production.
60  		
61  		MarkSet<Host> marks = map.get( failed );
62  		
63  		marks.mark( reporter );
64  		
65  		if ( marks.size() > config.getHosts().size() * QUORUM ) {
66  			
67  			log.warn( "Host is no longer online: %s (%,d hosts have gossiped that it has failed.)", failed , marks.size() );
68  			
69  			// we can't use this host anymore.  
70  			offline.mark( failed );
71  
72  			online.clear( failed );
73  			
74  		}
75  		
76  	}
77      
78  }
