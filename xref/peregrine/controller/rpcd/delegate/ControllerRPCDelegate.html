41      /**
42       * Allows a worker node to report that a partition is complete and its
43       * mapper/reducers have executed correctly.
44       */
45      @RPC
46      public ChannelBuffer complete( ControllerDaemon controllerDaemon, Channel channel, Message message )
47          throws Exception {
48  
49          Host host     = Host.parse( message.get( "host" ) );
50          Input input   = new Input( message.getList( "input" ) );
51          Work work     = new Work( host, input, message.getList( "work" ) );
52  
53          Scheduler scheduler = controllerDaemon.getScheduler();
54  
55          if ( scheduler != null )
56              scheduler.markComplete( host, work );
57          
58          return null;
59  		
60      }
61  	
62      /**
63       * Allows a worker node to report that a given partition has failed.  This
64       * is usually done if the machine is functioning correctly.  If not we mark
65       * the machine failed via other means (such as gossip).
66       */
67      @RPC
68      public ChannelBuffer failed( ControllerDaemon controllerDaemon, Channel channel, Message message )
69          throws Exception {
70          
71          Host host          = Host.parse( message.get( "host" ) );
72          Input input        = new Input( message.getList( "input" ) );
73          Work work          = new Work( host, input, message.getList( "work" ) );
74          String stacktrace  = message.get( "stacktrace" );
75          boolean killed     = message.getBoolean( "killed" );
76          
77          Scheduler scheduler = controllerDaemon.getScheduler();
78  
79          if ( scheduler != null )
80              scheduler.markFailed( host, work, killed, stacktrace );
81  	    
82          return null;
83  		
84      }

86      /**
87       * Allows a worker to tell a controller that a given chunk within a unit of
88       * work has been executed.  This could be a chunk in a map task, a chunk
89       * position list in a merge, or an individual sort in a preemptive reduce
90       * sort or even the final reduce sort.
91       */
92      @RPC
93      public ChannelBuffer progress( ControllerDaemon controllerDaemon, Channel channel, Message message )
94          throws Exception {
95          
96          Report report = new Report();
97          report.fromMessage( message.getMessage( "report" ) );
98  
99          Scheduler scheduler = controllerDaemon.getScheduler();
100 
101         if ( scheduler != null )
102             scheduler.updateReport( report );
103 
104         return null;
105 		
106     }

108     /**
109      * Allows the controller to receive 'heartbeats' from machines in the
110      * cluster so that it can obtain status on the cluster for which machines
111      * are 'out there' in the ether.
112      */
113     @RPC
114     public ChannelBuffer heartbeat( ControllerDaemon controllerDaemon, Channel channel, Message message )
115         throws Exception {
116         
117         Host host = Host.parse( message.get( "host" ) );
118         
119         // verify that the config_checksum is correct...
120         if ( ! controllerDaemon.getConfig().getChecksum().equals( message.get( "config_checksum" ) ) ) {
121             throw new Exception( "Config checksum from %s is invalid: " + host );
122         }
123         
124         // mark this host as online for the entire controller.
125         controllerDaemon.getClusterState().getOnline().mark( host );
126 		
127         return null;
128 		
129     }

131     /**
132      * Allows a worker node to report back that it failed to communicate /
133      * collaborate with a given host.  The controller then receives these
134      * messages and can make informed decisions about which to mark offline.
135      */
136     @RPC
137     public ChannelBuffer gossip( ControllerDaemon controllerDaemon, Channel channel, Message message )
138         throws Exception {
139         
140         // mark that a machine has failed to process some unit of work.
141         
142         Host reporter = Host.parse( message.get( "reporter" ) );
143         Host failed   = Host.parse( message.get( "failed" ) );
144         
145         controllerDaemon.getClusterState().getGossip().mark( reporter, failed ); 
146         
147         return null;
148 		
149     }

166     /**
167      * Take a batch message and hand it to the controller.
168      */
169     @RPC
170     public ChannelBuffer submit( ControllerDaemon controllerDaemon, Channel channel, Message message )
171         throws Exception {
172         
173         Batch batch = new Batch();
174         batch.fromMessage( new Message( message.get( "batch" ) ) );
175 
176         // submit the batch for execution
177         batch = controllerDaemon.getController().submit( batch );
178 
179         // the batch will now have an identifier and we should return it as part of the response
180 
181         Message result = new Message();
182         result.put( "identifier", batch.getIdentifier() );
183         
184         return result.toChannelBuffer();
185 		
186     }

188     /**
189      * Wait for a given batch job to complete and return successfuly once we
190      * have completed.
191      */
192     @RPC
193     public ChannelBuffer waitFor( ControllerDaemon controllerDaemon, Channel channel, Message message )
194         throws Exception {
195 
196         List<Batch> history = controllerDaemon.getController().getHistory();
197 
198         long identifier = message.getLong( "identifier" );
199         
200         if ( history.size() == 0 || history.get( 0 ).getIdentifier() > identifier ) {
201             throw new Exception( "Batch job not yet complete: " + identifier );
202         }
203 
204         Batch batch = null;
205 
206         for( Batch current : history ) {
207             if ( identifier == current.getIdentifier() ) {
208                 batch = current;
209                 break;
210             }
211         }
212 
213         if ( batch == null ) {
214             // this should be 404 not found right?
215             throw new Exception( "Unable to find batch with identifier: " + identifier );
216         }
217 
218 
219         // try the batch and return it.
220         Message result = new Message();
221 
222         result.put("name", batch.getName());
223         result.put("state", batch.getState());
224         result.put("started", batch.getStarted());
225         result.put("duration", batch.getDuration());
226         result.put("cause", batch.getCause());
227         
228         return result.toChannelBuffer();
229 
230     }
