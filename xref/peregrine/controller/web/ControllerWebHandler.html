41  /**
42   * Serve VTL web interface for the controller.
43   */
44  public class ControllerWebHandler extends SimpleChannelUpstreamHandler  {
45      
46      private static final Logger log = Logger.getLogger();
47  
48      protected Config config;
49      protected ControllerDaemon controllerDaemon;
50      protected String uri;
51      
52      public ControllerWebHandler( Config config,
53                                   ControllerDaemon controllerDaemon,
54                                   String uri ) {
55          
56          this.config = config;
57          this.controllerDaemon = controllerDaemon;
58          this.uri = uri;
59          
60      }
61      
62      @Override
63      public void messageReceived( ChannelHandlerContext ctx, MessageEvent e ) throws Exception {
64  
65          Channel channel = e.getChannel();
66  
67          VelocityContext context = new VelocityContext();
68          context.put( "config", config );
69          context.put( "controller", controllerDaemon.getController() );
70          context.put( "scheduler", controllerDaemon.getScheduler() );
71  
72          String basename = basename( uri, "index.vm" );
73  
74          QueryStringDecoder decoder = new QueryStringDecoder( uri );
75  
76          Map<String,List<String>> params = decoder.getParameters();
77  
78          // new map with just key/value mappings.  
79          Map<String,String> map = new HashMap<String,String>();
80  
81          for( String key : params.keySet() ) {
82              map.put( key, params.get( key ).get(0) );
83          }
84  
85          context.put( "params", map );
86          
87          StringWriter sw = new StringWriter();
88  
89          Template template = Velocity.getTemplate("web/" + basename);
90  
91          template.merge( context, sw );
92          
93          ChannelBuffer content = ChannelBuffers.wrappedBuffer( sw.toString().getBytes() );
94  
95          HttpResponse response = new DefaultHttpResponse( HTTP_1_1, OK );
96          response.setContent( content );
97          
98          channel.write(response).addListener(ChannelFutureListener.CLOSE);
99          
100     }
101 
102     // get the basename (file without directory) from the URI
103     public static String basename( String uri, String _default ) {
104 
105         // remove the query portion.
106         int idx = uri.indexOf( "?" );
107 
108         if ( idx >= 0 ) {
109             uri = uri.substring( 0, idx );
110         }
111 
112         idx = uri.lastIndexOf( "/" );
113 
114         if ( idx == -1 ) {
115             return _default;
116         }
117 
118         uri = uri.substring( idx + 1, uri.length() );
119 
120         if ( Strings.empty(uri) ) {
121             return _default;
122         }
123 
124         return uri;
125 
126     }
