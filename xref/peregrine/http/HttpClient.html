89      public HttpClient( HttpClientRuntime httpClientRuntime, HttpMethod method, String url ) throws peregrine.http.HttpClientException {
90  
91          Config config = httpClientRuntime.getConfig();
92          this.url = url;
93  
94          setConnectTimeout(config.getNetConnectTimeout());
95          setWriteTimeout( config.getNetWriteTimeout() );
96  
97          try {
98              URI uri = new URI( url );
99  
100             host = uri.getHost();
101             port = uri.getPort();
102 
103         } catch ( URISyntaxException e ) {
104             throw createHttpClientException(e);
105         }
106 
107         if ( port <= 0 )
108             port = 80;
109 
110         // Configure the client.
111         bootstrap = new BootstrapFactory( config ).newClientBootstrap( httpClientRuntime.getSocketChannelFactory() );
112 
113         // Set up the event pipeline factory.
114         bootstrap.setPipelineFactory( new peregrine.http.HttpClientPipelineFactory( config, this ) );
115 
116         setRequest( new DefaultHttpRequest( HttpVersion.HTTP_1_1, method, url ) );
117 
118     }

120     public void connect() throws peregrine.http.HttpClientException {
121 
122         // startup all latches here to allow users to override timeouts.
123         this.connectedLatch = new peregrine.http.HttpTrackingLatch( this, 1, getConnectTimeout() );
124         this.respondedLatch = new peregrine.http.HttpTrackingLatch( this, 1, getWriteTimeout() );
125         this.closedLatch = new peregrine.http.HttpTrackingLatch( this, 1, getConnectTimeout() );
126 
127         // these must be set every time so that we can reset them when using
128         // persistent connections.
129         this.cause = new AtomicReference<Throwable>( null );
130         this.response = new AtomicReference<HttpResponse>( null );
131 
132         // we need new content after each connect.
133         this.content = new ConcurrentLinkedQueue<ChannelBuffer>();
134 
135         request.setHeader( HttpHeaders.Names.TRANSFER_ENCODING, HttpHeaders.Values.CHUNKED );
136         request.setHeader( HttpHeaders.Names.USER_AGENT, getClass().getName() );
137 
138         // Start the connection attempt... where is the connect timeout set?
139         bootstrap.connect( new InetSocketAddress( host, port ) )
140                 .addListener( new ChannelFutureListener() {
141                     @Override
142                     public void operationComplete(ChannelFuture channelFuture) throws Exception {
143 
144                         if ( channelFuture.isSuccess() ) {
145 
146                             channel = channelFuture.getChannel();
147                             channel.write( request );
148 
149                             channel.getCloseFuture().addListener( new ChannelFutureListener() {
150                                 @Override
151                                 public void operationComplete(ChannelFuture future) throws Exception {
152                                     closedLatch.countDown();
153                                 }
154                             } );
155 
156                         } else {
157 
158                             // if we have an initial connection failure we should
159                             // immediately get a cause and NOT have to wait until
160                             // timeout.
161                             markCause( channelFuture.getCause() );
162 
163                         }
164 
165                         connectedLatch.countDown();
166 
167                     }
168                 } );
169 
170         connectedLatch.awaitUninterruptibly();
171 
172     }

186     /**
187      * Wait for an HTTP response from this channel.  Without necessarily
188      * closing it.  This can be used for connection reuse (AKA HTTP keep-alive).
189      * This is (by definition) a synchronous operation.
190      */
191     public void waitForResponse() throws peregrine.http.HttpClientException {
192         respondedLatch.awaitUninterruptibly();
193         assertSuccess();
194     }

196     public void close() throws peregrine.http.HttpClientException {
197 
198         if ( closed )
199             return;
200 
201         waitForResponse();
202 
203         if ( channel.isOpen() ) {
204             channel.close();
205         }
206 
207         closedLatch.awaitUninterruptibly();
208         assertSuccess();
209 
210         closed = true;
211 
212     }

304     protected void assertSuccess() throws peregrine.http.HttpClientException {
305 
306         if ( getCause() != null )
307             throw createHttpClientException(getCause());
308 
309         if ( getResponse() != null ) {
310 
311             if ( ! HttpResponseStatus.OK.equals( getResponse().getStatus() ) ) {
312                 throw createHttpClientException(null);
313             }
314 
315         }
316 
317     }

338     /**
339      *
340      * Create a new HttpClientException using the current client as the template.
341      */
342     protected peregrine.http.HttpClientException createHttpClientException( Throwable cause ) {
343 
344         int status = peregrine.http.HttpClientException.STATUS_TIMEOUT;
345 
346         if ( getResponse() != null )
347             status = getResponse().getStatus().getCode();
348 
349         return new peregrine.http.HttpClientException( cause, request.getMethod().getName(), url, status );
