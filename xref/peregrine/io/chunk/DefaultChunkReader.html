144     private void init( ChannelBuffer buff, StreamReader reader ) {
145 
146     	this.buffer = buff;
147         this.reader = reader;
148         this.length = buff.writerIndex();
149 
150         if ( readTrailer ) {
151 
152             //seek the to the end and read the magic ... 
153             assertLength(Integers.LENGTH);
154             int magic = buffer.getInt( buffer.writerIndex() - Integers.LENGTH );
155 
156             if ( magic >= 0 ) {
157 
158                 count = magic;
159                 minimal = true;
160                 
161             } else {
162 
163                 assertLength(Trailer.LENGTH);
164 
165                 trailer.read( buff );
166                 count = trailer.getCount();
167 
168                 // read the file info
169                 if ( trailer.getFileInfoOffset() > 0 ) {
170                     fileInfo.read( buff, trailer );
171                 }
172                 
173                 // read data and meta index
174 
175                 if ( trailer.getIndexOffset() > 0 ) {
176                     
177                     buff.readerIndex( (int)trailer.getIndexOffset() );
178 
179                     for( int i = 0; i < trailer.getIndexCount(); ++i ) {
180                         IndexBlock db = new IndexBlock();
181                         db.read( buff );
182                         indexBlocks.add( db );
183                     }
184 
185                 }
186 
187             }
188                 
189             // jump us back to the beginning of the buffer.
190             buff.readerIndex( 0 );
191 
192         }
193

276     /**
277      * Fetch all the keys in the given IndexBlock.
278      * @return A list of all BackendRequests that are not complete.
279      */
280     public List<BackendRequest> seekTo( List<BackendRequest> requests, IndexBlock block, RecordListener listener ) throws IOException {
281 
282         if ( requests.size() == 0 )
283             return requests;
284         
285         // position us at the beginning of the block.
286         buffer.readerIndex( block.getOffset() );
287 
288         regionMetrics.blocksRead.incr();
289 
290         int seek_idx = 0;
291 
292         SeekToContext context = new SeekToContext( listener, requests );
293 
294         // the request+key we should be looking for...
295         context.find = requests.remove( 0 );
296 
297         while( seek_idx < block.getCount()) {
298 
299             next();
300             ++seek_idx;
301 
302             regionMetrics.seekToKeyReads.incr();
303 
304             // TODO: skip suspended clients
305             //
306             // https://bitbucket.org/burtonator/peregrine/issue/210/sstable-make-sure-suspended-clients-arent
307 
308             context.handleScanRequests( key(), value() );
309 
310             context.handleCurrent( key(), value() );
311 
312             // terminate early if possible.
313             if ( context.isFinished() )
314                 break;
315 
316         }
317 
318         // at this point any remaining requests are NOT matched.  If it's a scan
319         // request it has an implicit start key (seek key) that wasn't matched
320         // and if it's a get request for a specific key then that key wasn't found
321         // in the routed data block.
322         for( BackendRequest request : requests ) {
323             request.setComplete(true);
324         }
325 
326         if ( context.find != null )
327             context.find.setComplete( true );
328 
329         // return any incomplete scan requests.  It does not make sense to return
330         // any get/fetch requests because they won't be continued in the next
331         // block
332         return context.partialScanRequests;
333 
334     }

336     // move some of the seek context information into a class to keep code and
337     // methods tight.
338     class SeekToContext {
339 
340         private RecordListener listener;
341 
342         private List<BackendRequest> requests;
343 
344         SeekToContext(RecordListener listener, List<BackendRequest> requests) {
345             this.listener = listener;
346             this.requests = requests;
347         }
348 
349         protected BackendRequest find = null;
350 
351         // keep a list of keys that haven't yet finished serving all records.
352         // these are going to be SCAN requests in practice.
353         List<BackendRequest> partialScanRequests = new ArrayList<BackendRequest>();
354 
355         // once we've found scan requests we have to listen to them until they
356         // are complete
357         protected void handleScanRequests( StructReader key,
358                                            StructReader value ) {
359 
360             Iterator<BackendRequest> scanIterator = partialScanRequests.iterator();
361 
362             while( scanIterator.hasNext() ) {
363 
364                 BackendRequest current = scanIterator.next();
365 
366                 current.visit( key, value );
367 
368                 if ( current.isFound() ) {
369                     listener.onRecord( current, key, value );
370                 }
371 
372                 if ( current.isComplete() ) {
373                     scanIterator.remove();
374                 }
375 
376             }
377 
378         }
379 
380         protected void handleCurrent( StructReader key, StructReader value ) {
381 
382             // find can be null during remaining scan requests.
383             if ( find != null ) {
384 
385                 while( true ) {
386 
387                     find.visit( key, value );
388 
389                     if ( find.isFound() ) {
390 
391                         listener.onRecord( find, key, value );
392 
393                         regionMetrics.seekToKeyHits.incr();
394 
395                         if ( find.isComplete() == false ) {
396                             // this is a scan request and we're not done yet.
397                             partialScanRequests.add(find);
398                         }
399 
400                     }
401 
402                     if ( find.hasSibling() && requests.size() > 0 ) {
403                         find = requests.remove( 0 );
404                         continue;
405                     }
406 
407                     break;
408 
409                 }
410 
411                 // either the key matched or we've gone PAST the key we are
412                 // searching for and we don't need to keep searching for this key.
413                 if ( find.isFound() || find.isComplete() ) {
414 
415                     if ( requests.size() > 0 ) {
416                         find = requests.remove( 0 );
417                     } else {
418                         // we've handled all the keys in this request so we can quit now.
419                         find = null;
420                         return;
421                     }
422 
423                 }
424 
425             }
426 
427         }
428 
429         protected boolean isFinished() {
430 
431             // we are complete if there aren't any pending scan requests AND
432             // there are no more requests to execute.
433             return find == null && partialScanRequests.size() == 0;
434 
435         }
436 
437     }
