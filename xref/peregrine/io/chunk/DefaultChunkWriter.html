38  /**
39   * <p>
40   * Write key/value pairs to a given file on disk and include any additional
41   * metadata (size, etc).
42   * </p>
43   *
44   * <p> The output format is very simple.  The file is a collection of key value
45   * pairs.  First read a varint.  This will denote how many bytes to read next.
46   * That will become your key.  Then read another varint, then read that many
47   * bytes and that will become your value.
48   * </p>
49   *
50   * <p>
51   * The on disk block format is similar to hbase and bigtable.  Records are written
52   * at the beginning of the table in series of implicit data blocks.  At the end
53   * of the table we then write index blocks which store the start key and offset
54   * of each data block.  After that we write a series of meta blocks which can
55   * contain key/value pairs.  The meta blocks stores information like the bloom
56   * filter index.
57   * </p>
58   *
59   * <p>
60   * https://bitbucket.org/burtonator/peregrine/issue/195/design-sstable-file-layout
61   * </p>
62   */
63  public class DefaultChunkWriter extends BaseSSTableChunk implements ChunkWriter {
64  
65      protected static final Logger log = Logger.getLogger();
66  
67      public static int BUFFER_SIZE = 16384 ;
68  
69      // default block size for writing indexed chunks
70      protected long blockSize = -1;
71  
72      // the main writable for all output.
73      protected ChannelBufferWritableWithLength writer = null;
74  
75      // true when closed.
76      private boolean closed = false;
77  
78      // true when shutdown.
79      private boolean shutdown = false;
80  
81      // the current IndexBlock
82      protected IndexBlock indexBlock = null;
83  
84      // the current MetaBlock
85      protected MetaBlock metaBlock = null;
86  
87      // the last key we saw in the whole stream.  This way we have the first key
88      // of each block as well as the last key of the entire file.
89      protected StructReader lastKey = null;
90  
91      // when true, skip the index and just write a minimal chunk stream.  Minimal
92      // chunks have a -1 at the end of the stream which would normally be invalid
93      // for a minimal stream since the last four bytes is the 'count'.  Consider
94      // having two implementations of this class each implementing different
95      // components (minimal and indexed).
96      protected boolean minimal = true;
97      
98      public DefaultChunkWriter( Config config, ChannelBufferWritable writer ) throws IOException {
99          init( config, writer );
100     }

102     public DefaultChunkWriter( Config config, File file ) throws IOException {
103         init( config, new MappedFileWriter( config, file ) );

106     private void init( Config config, ChannelBufferWritable writer ) {
107 
108         if ( config == null )
109             throw new NullPointerException( "config" );
110 
111         if ( writer instanceof BufferedChannelBufferWritable ) {
112 
113             // don't attempt to double buffer.
114             this.writer = (ChannelBufferWritableWithLength)writer;
115 
116         } else if ( writer instanceof BackedChannelBufferWritable ) {
117 
118             // this is a writable backed with a ChannelBuffer so again, no
119             // need to buffer.
120             this.writer = (ChannelBufferWritableWithLength)writer;
121 
122         } else {
123 
124             // this is either a network based writer or a disk writer and we
125             // should buffer.
126             this.writer = new BufferedChannelBufferWritable( writer, BUFFER_SIZE );
127 
128         }
129 
130         this.blockSize = config.getSSTableBlockSize();
131 
132     }

150     @Override
151     public void write( StructReader key, StructReader value )
152         throws IOException {
153 
154         if ( closed )
155             throw new IOException( "closed" );
156 
157         if ( indexBlock == null || writer.length() - indexBlock.getOffset() > blockSize ) {
158             rollover( key );
159         }
160 
161         write( writer, key, value );
162 
163         trailer.incrRecordUsage( key.length() );
164 
165         if ( value != null ) {
166             trailer.incrRecordUsage( value.length() );
167         }
168 
169         trailer.incrCount();
170         indexBlock.incrCount();
171         lastKey = key;
172 
173     }

175     /**
176      * Perform a DIRECT write on a ChannelBuffer of a key/value pair.
177      */
178     public static int write( ChannelBufferWritable writer ,
179                              StructReader key,
180                              StructReader value ) throws IOException {
181 
182     	int result = 0;
183 
184         // TODO: we have to use an atomic write so that the buffered writer can
185         // make sure to get one key/value pair without truncating it
186         // incorrectly.
187         
188         ChannelBuffer wrapped = ChunkEntry.encode( key );
189 
190         // the beginning of
191         if ( value != null ) {
192             wrapped = wrappedBuffer( wrapped, ChunkEntry.encode( value ) );
193         }
194 
195         writer.write( wrapped );
196         
197         result += wrapped.writerIndex();
198 
199         return result;
200         
201     }
202 
203     private void rollover( StructReader key) {
204         endBlock();
205         startIndexBlock(key);
206     }
207 
208     private void endBlock() {
209 
210         if ( indexBlock != null && indexBlock.getCount() != 0 ) {
211             endIndexBlock();
212             addMetaBlockToIndex();
213         }
214 
215     }
216 
217     // create a new data block and add it to the data block list and return the
218     // newly created block.
219     private void startIndexBlock(StructReader key) {
220 
221         // create a new index block with teh correct first key specified.
222         indexBlock = new IndexBlock( key.toByteArray() );
223         indexBlock.setOffset(writer.length());
224 
225         indexBlocks.add(indexBlock);
226 
227     }
228 
229     private void endIndexBlock() {
230 
231         //TODO: compression would go here.
232         indexBlock.setLength(writer.length() - indexBlock.getOffset());
233         indexBlock.setLengthUncompressed(indexBlock.getLength());
234     }
235 
236     private void startMetaBlock() {
237 
238         metaBlock = new MetaBlock();
239 
240         metaBlocks.add( metaBlock );
241         
242     }
243 
244     private void endMetaBlock() {
245     }
246 
247     // write out a new meta block file which would include bloom filter data as
248     // well as any other additional metadata.
249     private void addMetaBlockToIndex() {
250 
251         startMetaBlock();
252         
253         //noop right now.  We don't have any metadata to write.
254 
255         endMetaBlock();
256         
257     }

259     /**
260      * Perform all final operations before we actually close the writer.
261      */
262     public void shutdown() throws IOException {
263 
264         if ( shutdown )
265             return;
266         
267         // write trailer to store the number of items.
268 
269         trailer.setDataSectionLength(writer.length());
270         
271         endBlock();
272 
273         if ( minimal ) {
274 
275             writer.write( StructReaders.wrap( trailer.getCount() )
276                               .getChannelBuffer() );
277             
278         } else {
279 
280             // write out file info
281             if ( lastKey != null ) {
282                 fileInfo.setLastKey( lastKey.toByteArray() );
283             }
284 
285             trailer.setFileInfoOffset(writer.length());
286             fileInfo.write( writer );
287 
288             for( int i = 0; i < metaBlocks.size(); ++i ) {
289 
290                 IndexBlock ib = indexBlocks.get(i);
291                 MetaBlock mb = metaBlocks.get(i);
292 
293                 ib.setMetaBlockOffset( (int)writer.length() );
294                 mb.write( writer );
295                 ib.setMetaBlockLength( (int) ( writer.length() - ib.getMetaBlockOffset() ) );
296 
297             }
298 
299             trailer.setIndexOffset(writer.length());
300             trailer.setIndexCount(indexBlocks.size());
301 
302             for( IndexBlock ib : indexBlocks) {
303                 ib.write( writer );
304             }
305 
306             // write the trailer
307             trailer.write( writer );
308         
309         }
310 
311         writer.shutdown();
312         
313         shutdown = true;
314         
315     }
