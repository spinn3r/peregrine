34  /**
35   * A RollingSequenceWriter that stores data in direct memory buffers where we
36   * use a DefaultChunkWriter and DefaultChunkReader to append to an in-memory
37   * index using 1MB slabs of direct memory.  When that slab is exhausted we
38   * create a new slab.  The keys are actually varints, the first varint is for
39   * the slab index and the second varint is for the offset within the slab.
40   * Since they  are direct buffers I can send them directly out over the kernel
41   * via netty (though I should test to make sure this works via zero copy).  In
42   * fact, the system we use whereby we allocate new chunks on rollover is VERY
43   * similar to DefaultPartitionWriter except instead of new files we create new
44   * slabs.  Though DefaultPartitionWriter has more complexity than we really need
45   * but the general theory is the same.
46   */
47  public class ArenaRollingSequenceWriter extends DirectBufferRollingSequenceWriter {
48  
49      public ArenaRollingSequenceWriter(Config config ) {
50          super( config, config.getMemtableArenaChunkSize(), true );
51      }
52  
53      public StructReader getValue(ArenaRecord ptr) {
54  
55          DefaultChunkReader reader = readers.get( ptr.getArena() );
56          Record record = reader.read( ptr.getOffset() );
57  
58          return record.getKey();
59  
60      }
61  
62      /**
63       * Get an ArenaRecord pointer for the last value written.
64       */
65      public ArenaRecord getArenaRecordForLastWrite() {
66  
67          // cast this to an int.. we're NEVER going to use > 2GB slabs
68          return new ArenaRecord( getCurrentID(), (int)getRecordOffsetForLastWrite() );
69  
70      }
71  
72      /**
73       * Get the number of direct buffers backing this arena.
74       * @return
75       */
76      public int getDirectBufferCount() {
77          return readers.size();
78      }
79  
80  }
