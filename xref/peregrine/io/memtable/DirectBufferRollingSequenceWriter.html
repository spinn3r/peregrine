17  package peregrine.io.memtable;
18  
19  import org.jboss.netty.buffer.ChannelBuffer;
20  import peregrine.Record;
21  import peregrine.config.Config;
22  import peregrine.io.SequenceWriter;
23  import peregrine.io.buffer.DirectMemoryAllocator;
24  import peregrine.io.buffer.DirectMemoryRegion;
25  import peregrine.io.chunk.DefaultChunkReader;
26  import peregrine.io.chunk.DefaultChunkWriter;
27  import peregrine.io.sstable.RollingSequenceWriter;
28  import peregrine.util.DirectMemoryAllocationTracker;
29  import peregrine.util.netty.BackedChannelBufferWritable;
30  
31  import java.io.IOException;
32  import java.util.ArrayList;
33  import java.util.List;
34  
35  /**
36   * A RollingSequenceWriter that stores data in direct memory buffers where we
37   * use a DefaultChunkWriter and DefaultChunkReader to append to an in-memory
38   * index using 1MB slabs of direct memory.  When that slab is exhausted we
39   * create a new slab.  The keys are actually varints, the first varint is for
40   * the slab index and the second varint is for the offset within the slab.
41   * Since they  are direct buffers I can send them directly out over the kernel
42   * via netty (though I should test to make sure this works via zero copy).  In
43   * fact, the system we use whereby we allocate new chunks on rollover is VERY
44   * similar to DefaultPartitionWriter except instead of new files we create new
45   * slabs.  Though DefaultPartitionWriter has more complexity than we really need
46   * but the general theory is the same.
47   */
48  public class DirectBufferRollingSequenceWriter extends RollingSequenceWriter {
49  
50      //keep track of allocated readers so that we can use them via getValue
51      protected List<DefaultChunkReader> readers = new ArrayList<DefaultChunkReader>();
52  
53      protected List<DirectMemoryRegion> directMemoryRegions = new ArrayList<DirectMemoryRegion>();
54  
55      private Config config;
56  
57      private long chunkSize;
58  
59      private boolean keyOnly;
60  
61      private DirectMemoryAllocator allocator;
62  
63      public DirectBufferRollingSequenceWriter(Config config, long chunkSize, boolean keyOnly ) {
64          super( chunkSize );
65  
66          this.config = config;
67          this.chunkSize = chunkSize;
68          this.keyOnly = keyOnly;
69          this.allocator = new DirectMemoryAllocator( config.getDirectMemoryAllocationTracker() );
70      }
71  
72      /**
73       * Get the current ID of the current block of memory for the last write.
74       */
75      public int getCurrentID() {
76          return readers.size() - 1;
77      }
78  
79      /**
80       * Get the number of direct buffers backing this arena.
81       * @return
82       */
83      public int getDirectBufferCount() {
84          return readers.size();
85      }
86  
87      public Record read(ArenaRecord ptr) {
88  
89          DefaultChunkReader reader = readers.get( ptr.getArena() );
90          return reader.read( ptr.getOffset() );
91  
92      }
93  
94      @Override
95      protected SequenceWriter newSequenceWriter() throws IOException {
96  
97          DirectMemoryRegion directMemoryRegion = allocator.allocate( (int)chunkSize );
98  
99          ChannelBuffer buff = directMemoryRegion.getBuffer();
100 
101         BackedChannelBufferWritable writable = new BackedChannelBufferWritable( buff );
102 
103         DefaultChunkReader reader = new DefaultChunkReader( buff, false );
104         reader.setKeyOnly( keyOnly );
105         readers.add( reader );
106         directMemoryRegions.add( directMemoryRegion );
107 
108         DefaultChunkWriter result = new DefaultChunkWriter( config, writable );
109         result.setMinimal( true );
110 
111         return result;
112 
113     }
114 
115     @Override
116     public void close() throws IOException {
117         super.close();
118 
119         // free up all buffers after this is closed.
120         for( DirectMemoryRegion region : directMemoryRegions ) {
121             region.close();
122         }
123 
124     }
125 }
