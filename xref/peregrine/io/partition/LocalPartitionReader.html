115     public LocalPartitionReader( Config config,
116                                  Partition partition,
117                                  String path,
118                                  List<ChunkStreamListener> listeners ) throws IOException {
119 
120         this.partition = partition;
121         this.chunkReaders = LocalPartition.getChunkReaders( config, partition, path );
122         this.iterator = chunkReaders.iterator();
123         this.path = path;
124 
125         this.chunkRef = new ChunkReference( partition, path );
126 
127         addListeners( listeners );
128 
129         // update the count by looking at all the chunks and returning the number
130         // of key/value pairs they contain.
131         for( DefaultChunkReader reader : chunkReaders ) {
132             count += reader.count();
133         }
134 
135         indexBlockLookup = buildIndexBlockLookup();

139     private TreeMap<StructReader,IndexBlockReference> buildIndexBlockLookup() {
140 
141         TreeMap<StructReader,IndexBlockReference> result = new TreeMap( new StrictStructReaderComparator() );
142 
143         // build the index of the ChunkReaders.
144 
145         byte[] lastKey = null;
146 
147         IndexBlockReference last = null;
148 
149         for ( int idx = 0; idx < chunkReaders.size(); ++idx ) {
150 
151             DefaultChunkReader current = chunkReaders.get( idx );
152             
153             for( IndexBlock db : current.getIndexBlocks() ) {
154 
155                 if ( firstKey == null )
156                     firstKey = StructReaders.wrap(db.getFirstKey());
157 
158                 IndexBlockReference ref = new IndexBlockReference();
159                 ref.reader = current;
160                 ref.idx = idx;
161                 ref.indexBlock = db;
162 
163                 result.put( StructReaders.wrap( db.getFirstKey() ), ref );
164 
165                 // update the linked list structure
166                 if ( last != null ) {
167                     last.next = ref;
168                 }
169 
170                 last = ref;
171 
172             }
173 
174             if ( current.getTrailer().getCount() > 0 ) {
175                 lastKey = current.getFileInfo().getLastKey();
176             }
177             
178         }
179 
180         // this is a fake entry for the last key so that keys that are clearly
181         // past the index don't even show up in searches.
182         if ( lastKey != null ) {
183         
184             BigInteger ptr = new BigInteger( lastKey );
185             ptr = ptr.add( BigInteger.valueOf( 1 ) );
186             
187             byte[] pd = ptr.toByteArray();
188             
189             // NOTE: BigInteger is conservative with padding so we have
190             // to add any byte padding back in.
191             if ( pd.length < lastKey.length ) {
192                 
193                 byte[] tmp = new byte[lastKey.length];
194                 System.arraycopy( pd, 0, tmp, lastKey.length - pd.length, pd.length );
195                 pd = tmp;
196                 
197             }
198             
199             result.put( new StructReader( pd ), null );
200 
201         }
202 
203         return result;
204                     
205     }

209      * so that I can find it within the given DefaultChunkReader.

211     protected IndexBlockReference findIndexBlockReference(StructReader key) {
212         
213         Map.Entry<StructReader,IndexBlockReference> entry = indexBlockLookup.floorEntry(key);
214 
215         if ( entry == null )
216             return null;
217 
218         return entry.getValue();
219 
220     }

230     @Override
231     public Record seekTo( BackendRequest request ) throws IOException {
232 
233         List<BackendRequest> requests = new ArrayList();
234         requests.add( request );
235 
236         LastRecordListener lastRecordListener = new LastRecordListener( null );
237 
238         seekTo( requests, lastRecordListener );
239 
240         if ( lastRecordListener.getLastRecord() != null ) {
241             return lastRecordListener.getLastRecord();
242         }
243 
244         return null;

248     @Override
249     public boolean seekTo( List<BackendRequest> requests, RecordListener listener ) throws IOException {
250 
251         this.key = null;
252         this.value = null;
253 
254         // the list must be sorted before we service it so that we can
255         // access keys on the same block without going backwards and
256         // accessing the same block again.  A seek is involved with fetching
257         // a block as is optionally decompression and that's expensive.  By
258         // first sorting the set we avoid both expensive operations.
259         Collections.sort( requests );
260 
261         Map<IndexBlockReference,List<BackendRequest>> requestIndexBlockTable =
262                 new TreeMap<IndexBlockReference, List<BackendRequest>>();
263 
264         LastRecordListener lastRecordListener = new LastRecordListener( listener );
265         
266         for( BackendRequest request : requests ) {
267 
268             // TODO: skip suspended clients
269             //
270             // https://bitbucket.org/burtonator/peregrine/issue/210/sstable-make-sure-suspended-clients-arent
271 
272             if ( request instanceof ScanBackendRequest ) {
273 
274                 ScanBackendRequest scanBackendRequest = (ScanBackendRequest)request;
275 
276                 // We have to set the seek key as the first entry in this SSTable
277                 // so that we can seekTo it similar to the way GetBackendRequest
278                 // works.
279                 if ( scanBackendRequest.getSeekKey() == null ) {
280                     StructReader firstKey = getFirstKey();
281                     scanBackendRequest.setImplicitStartKey( firstKey );
282                 }
283 
284             }
285 
286             StructReader key = request.getSeekKey();
287 
288             IndexBlockReference ref = findIndexBlockReference(key);
289 
290             // this key isn't indexed so no need to check.
291             if ( ref == null ) {
292                 request.setComplete(true);
293                 continue;
294             }
295 
296             List<BackendRequest> requestsForReference = requestIndexBlockTable.get( ref );
297 
298             if ( requestsForReference == null ) {
299                 requestsForReference = new ArrayList();
300                 requestIndexBlockTable.put(ref, requestsForReference);
301             }
302 
303             requestsForReference.add( request );
304             
305         }
306 
307         // keep the references we need to page through.
308         List<IndexBlockReference> indexBlockReferences = new ArrayList<IndexBlockReference>();
309         indexBlockReferences.addAll(requestIndexBlockTable.keySet());
310 
311         while( indexBlockReferences.size() > 0 ) {
312 
313             IndexBlockReference ref = indexBlockReferences.remove(0);
314 
315             chunkReader = ref.reader;
316 
317             // set the iterator to ALL chunkReader including the current one and
318             // continue forward.
319             iterator = chunkReaders.subList( ref.idx, chunkReaders.size() ).iterator();
320 
321             // update the chunk ref that we're working with.
322             chunkRef = new ChunkReference( partition, ref.idx );
323 
324             List<BackendRequest> refKeys = requestIndexBlockTable.get( ref );
325 
326             List<BackendRequest> partialScanRequests =
327                     ref.reader.seekTo( refKeys, ref.indexBlock, lastRecordListener );
328 
329             if ( ref.next != null ) {
330 
331                 if ( partialScanRequests.size() > 0 ) {
332 
333                     // scan requests need to index multiple blocks
334                     refKeys = requestIndexBlockTable.get( ref.next );
335 
336                     if ( refKeys != null ) {
337                         refKeys.addAll( partialScanRequests );
338                         Collections.sort( refKeys );
339                     } else {
340                         requestIndexBlockTable.put(ref.next, partialScanRequests);
341                         indexBlockReferences.add( 0, ref.next );
342                     }
343 
344                 }
345 
346             } else  {
347 
348                 // this is the LAST block in the entire index and by definition there
349                 // is nothing left to index.
350 
351                 for( BackendRequest request : partialScanRequests ) {
352 
353                     request.setComplete(true);
354 
355                     //TODO: aren't these ALWAYS ScanBackendRequests?
356                     if ( request instanceof ScanBackendRequest ) {
357                         ScanBackendRequest scanBackendRequest = (ScanBackendRequest)request;
358                         scanBackendRequest.setRegionExhausted( true );
359                     }
360 
361                 }
362 
363             }
364 
365         }
366 
367         if ( lastRecordListener.getLastRecord() != null ) {
368 
369             Record last = lastRecordListener.getLastRecord();
370             
371             this.key = last.getKey();
372             this.value = last.getValue();
373 
374             return true;
375             
376         } else {
377             return false;
378         }

386     @Override
387     public boolean hasNext() throws IOException {
388 
389         if ( chunkReader != null )
390             hasNext = chunkReader.hasNext();
391 
392         if ( hasNext == false ) {
393 
394             fireOnChunkEnd( chunkRef );
395             
396             if ( iterator.hasNext() ) {
397 
398                 chunkRef.incr();
399                 
400                 chunkReader = iterator.next();
401 
402                 fireOnChunk( chunkRef );
403                 
404                 hasNext = chunkReader.hasNext();
405 
406             } else {
407                 hasNext = false;
408             }
409 
410         }
411 
412         return hasNext;
413         
414     }

491     // contains a data block and a chunk reference and all the metadata needed
492     // to find a data block and seekTo the position of a key and optionally
493     // scan.
494     class IndexBlockReference implements Comparable<IndexBlockReference> {
495 
496         StrictStructReaderComparator comparator = new StrictStructReaderComparator();
497 
498         // keeps a pointer to the chunk this data block is stored 
499         protected DefaultChunkReader reader = null;
500 
501         // the position in the main chunk reader list so that we can call
502         // subList and get a new iterator for this and all subsequent chunks.
503         protected int idx = -1;
504 
505         // the data block for this reference.  Used so that we can call seekTo
506         // within the actual chunk.
507         protected IndexBlock indexBlock = null;
508 
509         // the next data block reference. We keep a forward linked list so that
510         // we can jump to the next block if a scan request isn't finished.
511         protected IndexBlockReference next = null;
512 
513         public int compareTo( IndexBlockReference db ) {
514 
515             StructReader key0 = StructReaders.wrap( indexBlock.getFirstKey() );
516             StructReader key1 = StructReaders.wrap( db.indexBlock.getFirstKey() );
517 
518             return comparator.compare( key0, key1 );
519 
520         }
521         
522     }
