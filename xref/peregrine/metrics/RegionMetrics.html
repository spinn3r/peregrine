19  import peregrine.StructReaders;
20  import peregrine.config.Partition;
21  import peregrine.io.SequenceWriter;
22  import peregrine.metrics.impl.CounterMetric;
23  import peregrine.metrics.impl.Metric;
24  import peregrine.util.Base64;
25  import peregrine.util.Longs;
26  
27  import java.io.IOException;
28  import java.util.Map;
29  
30  /**
31   * Metrics about a specific region we are hosting.
32   */
33  public class RegionMetrics extends MetricMap {
34  
35      // the partition for this region metric.  If the partition is null this is
36      // global for the entire system.
37      private Partition partition = null;
38  
39      public RegionMetrics() {
40          init(this);
41      }
42  
43      public RegionMetrics(Partition partition) {
44          init(this);
45          this.partition = partition;
46      }
47  
48      // clients that have been suspended and not keeping up with content.
49      public CounterMetric suspendedClients = new CounterMetric();
50  
51      public CounterMetric cancelledClients = new CounterMetric();
52  
53      public CounterMetric getRequests = new CounterMetric();
54  
55      public CounterMetric scanRequests = new CounterMetric();
56  
57      // the number of blocks read from disk.  Technically we don't read the
58      // whole block off disk as we quit once we've read all keys.
59      public CounterMetric blocksRead = new CounterMetric();
60  
61      public CounterMetric blocksWritten = new CounterMetric();
62  
63      public CounterMetric recordsRead = new CounterMetric();
64  
65      public CounterMetric recordsWritten = new CounterMetric();
66  
67      // the number of reads while searching for keys.
68      public CounterMetric seekToKeyReads = new CounterMetric();
69  
70      // the number of hits while searching for keys
71      public CounterMetric seekToKeyHits = new CounterMetric();
72  
73      public CounterMetric batchExecutions = new CounterMetric();
74  
75      // the number of times the back request queue was exhausted.  Clients write
76      // requests into the
77      public CounterMetric batchRequestQueueExhausted = new CounterMetric();
78  
79      // the number of times a memtable has been flushed
80      public CounterMetric memtablesFlushed = new CounterMetric();
81  
82      public CounterMetric compactions = new CounterMetric();
83  
84      public void toSequenceWriter( SequenceWriter writer ) throws IOException {
85  
86          for( Map.Entry<String,Metric> entry : entrySet() ) {
87  
88              String name = entry.getKey();
89              Metric metric = entry.getValue();
90  
91              String key = null;
92  
93              if ( partition == null ) {
94                  key = "worker." + name;
95              } else {
96                  key = "region." + Base64.encode(Longs.toByteArray(partition.getId())) + "." + name;
97              }
98  
99              writer.write( StructReaders.wrap( key ), StructReaders.wrap( Long.toString( metric.get() ) ) );
100 
101         }
102 
103     }
104 
105 }
