16  package peregrine.os;
17  
18  import com.spinn3r.log5j.Logger;
19  import com.sun.jna.Pointer;
20  import peregrine.util.IdempotentCloser;
21  
22  import java.io.Closeable;
23  import java.io.File;
24  import java.io.FileDescriptor;
25  import java.io.IOException;
26  
27  /**
28   * Facade on top of mlock/mmap that opens up a file, mmaps it, then mlocks it
29   * (optionally).
30   */
31  public class FileMapper extends IdempotentCloser implements Closeable {
32  
33      private static final Logger log = Logger.getLogger();
34  
35      private Pointer pa;
36      private long length;
37      private File file;
38      private FileDescriptor descriptor;
39  
40      private boolean lock = false;
41  
42      /**
43       * Call mmap a file descriptor, then lock the pages with MAP_LOCKED.  This
44       * will then prevent the pages from being swapped out due to VFS cache
45       * pressure.
46       * 
47       * @param descriptor The file we should mmap and MAP_LOCKED
48       * @param offset
49       * @param length
50       * @see #close()
51       * @throws IOException
52       */
53      public FileMapper( File file, FileDescriptor descriptor, long offset, long length ) throws IOException {
54  
55      	log.debug( "mapping %s with length %,d", file, length );
56  
57          this.file = file;
58      	this.descriptor = descriptor;
59          this.length = length;
60          
61          int fd = Platform.getFd( descriptor );
62  
63          int prot   = mman.PROT_READ;
64          int flags  = mman.MAP_SHARED;
65  
66          if ( lock ) {
67              flags  = mman.MAP_SHARED | mman.MAP_LOCKED;
68              log.debug( "mapping(+lock) %s with length %,d", file, length );
69          }
70  
71          pa = mman.mmap( length, prot, flags, fd, offset );
72  
73          // even though technically we have specified MAP_LOCKED this isn't
74          // supported on OpenSolaris or older Linux kernels (or OS X) so call
75          // mlock as well.
76  
77          if ( lock ) {
78              mman.mlock( pa, length );
79          }
80          
81      }
82  
83      public long getAddress() {
84          return Pointer.nativeValue( pa );
85      }
86  
87      /**
88       * Unlock a region of the file from zero to 'length' bytes without unlocking
89       * the rest of the file.  This can be used to unlock pages we've already
90       * read so the memory can be reclaimed by OS.
91       */
92      public void unlockRegion( long len ) throws IOException {
93          
94          mman.munlock( pa, len );
95          
96      }
97      
98      /**
99       * Release this lock so that the memory can be returned to the OS if it
100      * wants to us it.
101      */
102     @Override
103     protected void doClose() throws IOException {
104 
105         if ( lock ) {
106             log.debug( "munlocking %s to pa %s with length %,d", file, pa, length );
107             mman.munlock( pa, length );
108         }
109             
110         mman.munmap( pa, length );
111 
112     }
113 
114     public boolean getLock() { 
115         return this.lock;
116     }
117 
118     public void setLock( boolean lock ) { 
119         this.lock = lock;
120     }
121 
122 }
