19  import org.jboss.netty.buffer.ChannelBuffer;

23  import java.io.Closeable;
24  import java.io.File;
25  import java.io.FileOutputStream;
26  import java.io.IOException;

28  /**
29   * Facade around a MappedByteBuffer but we also support mlock on the mapped
30   * pages, and closing all dependent resources.
31   *
32   */
33  public class MappedFileWriter extends BaseMappedFile implements Closeable, ChannelBufferWritable {
34  
35      private static final Logger log = Logger.getLogger();
36  
37      public static boolean DEFAULT_AUTO_SYNC = true;
38  
39      protected FileOutputStream out;
40  
41      protected boolean autoSync = DEFAULT_AUTO_SYNC;
42  
43      protected long fallocateExtentSize = 0;
44  
45      protected long syncWriteSize = 0;
46  
47      /**
48       * Number of bytes we've allocated via fallocate.
49       */
50      long allocated = 0;
51  
52      /**
53       * The number of bytes we have written out to disk (aligned on a 4k page
54       * boundary for now).
55       */
56      long synced = 0;
57  
58      public MappedFileWriter( Config config, String path ) throws IOException {
59          this( config, new File( path ) );
60      }
61      
62      public MappedFileWriter( Config config, File file ) throws IOException {
63  
64          init( config, file );
65  
66      }
67  
68      private void init( Config config, File file) throws IOException {
69  
70          this.config = config;
71          this.file = file;
72  
73          if ( config != null ) {
74  
75              fadviseDontNeedEnabled  = config.getFadviseDontNeedEnabled();
76              fallocateExtentSize     = config.getFallocateExtentSize();
77              syncWriteSize           = config.getSyncWriteSize();
78              
79          }
80  
81          this.out = new FileOutputStream( file );
82          this.fd = Platform.getFd( out.getFD() );
83          this.channel = out.getChannel();
84  
85          this.length = file.length();
86  
87      }
88  
89      public boolean getAutoSync() { 
90          return this.autoSync;
91      }
92  
93      public void setAutoSync( boolean autoSync ) { 
94          this.autoSync = autoSync;
95      }
96  
97      @Override
98      protected void doClose() throws IOException {
99  
100         if ( closer.isClosed() )
101             return;
102 
103         shutdown();
104         
105         closer.add( channel );
106         closer.add( out );
107 
108         closer.close();
109 
110     }
111 
112     @Override
113     public void write( ChannelBuffer buff ) throws IOException {
114 
115         if ( closer.isClosed() ) {
116             throw new IOException( "closed" );
117         }
118         
119         long newLength = length + buff.writerIndex();
120 
121         // see if we're about to write past the previous fallocate extent
122         // size.
123         
124         if ( newLength > allocated && fallocateExtentSize > 0 ) {
125             
126             fcntl.posix_fallocate( fd, allocated, fallocateExtentSize );
127 
128             allocated += fallocateExtentSize;
129             
130         }
131 
132         if ( autoSync && syncWriteSize > 0 && newLength > synced + syncWriteSize ) {
133 
134             // slice the channel buffer into smaller regions which are page
135             // aligned and then write these aligned pages then sync when the
136             // pages are aligned.  Otherwise we would write a page 2x
137             // because we first do a partial write, then when it is dirtied
138             // again we re-write the same page.
139             
140             // the extra data that would be written.
141             long extra = newLength % syncWriteSize;
142 
143             /*
144              * 
145              * This is a visual diagram of the page alignment.  Each
146              * character is a 512 byte sector to shorten the visual
147              * representation.
148              * 
149              * bytes written:   |===========|
150              * page alignment:  |==============|
151              * the write:                    |=====|
152              */
153             
154             if ( extra > 0 ) {
155                 
156                 // the boundary between page aligned and extra data for THIS
157                 // buffer.
158                 int boundary = (int)(buff.writerIndex() - extra);
159 
160                 // write the filled page BEFORE the extra data... 
161                 write0( buff.readSlice( boundary ) );
162 
163                 // now update the buffer so that we write the data after the
164                 // current filled page before we exit.
165                 
166                 buff = buff.readSlice( buff.writerIndex() - boundary );
167                 
168             }
169             
170             sync();
171 
172         }
173 
174         write0( buff );
175         
176     }
177 
178     /**
179      * Write the entire ChannelBuffer RAW and update the length of bytes
180      * we've written (and perform no other operations) to the current
181      * channel.
182      */
183     private void write0( ChannelBuffer buff ) throws IOException {
184         buff.getBytes( 0, channel, buff.writerIndex() );
185         length += buff.writerIndex();
186     }
187     
188     @Override
189     public void shutdown() throws IOException {
190 
191         if ( autoSync && syncWriteSize > 0 ) {
192             sync();
193         }
194 
195         // if we're in fallocate mode, we now need to truncate the file so
196         // that it is the correct length.
197 
198         if ( fallocateExtentSize > 0 ) {
199             channel.truncate( length );
200         }
201 
202     }
203 
204     @Override
205     public void sync() throws IOException {
206         
207         channel.force( false );
208 
209         // TODO: I'm not sure this is the right strategy for ALL writes
210         // ... temporary files should probably ALL be evicted but writes may
211         // benefit if in situations where a box isn't tuned perfectly but it
212         // doesn't make sense to optimize for poorly configured machines.
213 
214         /*
215         if ( fadviseDontNeedEnabled ) {
216             fcntl.posix_fadvise( fd, offset, length, fcntl.POSIX_FADV_DONTNEED );
217         }
218         */
219 
220         synced = length;
221 
222     }
223 
224     @Override
225     public void flush() throws IOException {
226         sync();
227     }
