60      public enum RLIMIT {
61  
62          /**
63           * The maximum size of the process's virtual memory (address space) in
64           * bytes. This limit affects calls to brk(2), mmap(2) and mremap(2), which
65           * fail with the error ENOMEM upon exceeding this limit. Also automatic
66           * stack expansion will fail (and generate a SIGSEGV that kills the process
67           * if no alternate stack has been made available via sigaltstack(2)). Since
68           * the value is a long, on machines with a 32-bit long either this limit is
69           * at most 2 GiB, or this resource is unlimited.
70           */
71          AS(9,5)       ,
72  
73          /**
74           * The maximum number of bytes of memory that may be locked into RAM. In
75           * effect this limit is rounded down to the nearest multiple of the system
76           * page size. This limit affects mlock(2) and mlockall(2) and the mmap(2)
77           * MAP_LOCKED operation. Since Linux 2.6.9 it also affects the shmctl(2)
78           * SHM_LOCK operation, where it sets a maximum on the total bytes in shared
79           * memory segments (see shmget(2)) that may be locked by the real user ID of
80           * the calling process. The shmctl(2) SHM_LOCK locks are accounted for
81           * separately from the per-process memory locks established by mlock(2),
82           * mlockall(2), and mmap(2) MAP_LOCKED; a process can lock bytes up to this
83           * limit in each of these two categories. In Linux kernels before 2.6.9,
84           * this limit controlled the amount of memory that could be locked by a
85           * privileged process. Since Linux 2.6.9, no limits are placed on the amount
86           * of memory that a privileged process may lock, and this limit instead
87           * governs the amount of memory that an unprivileged process may lock.
88           */
89          MEMLOCK(8,6)  ,
90  
91          /**
92           * Specifies a value one greater than the maximum file descriptor number
93           * that can be opened by this process. Attempts (open(2), pipe(2), dup(2),
94           * etc.) to exceed this limit yield the error EMFILE. (Historically, this
95           * limit was named RLIMIT_OFILE on BSD.)
96           */
97          NOFILE(7,8)   ;
98  
99          private final int value;
100             
101         RLIMIT( int value_linux, int value_darwin ) {
102 
103             if ( Platform.isLinux() ) {
104                 value = value_linux;
105             } else if ( Platform.isDarwin() ) {
106                 value = value_darwin;
107             } else {
108                 throw new RuntimeException( "Unknown platform: " + Platform.getOS() );
109             }
110             
111         }
112 
113         public int value() {
114             return value;
115         }
116         
117     }

119     public static class RlimitStruct extends Structure {
120 
121         public RlimitStruct() {};
122         
123         public RlimitStruct( long value ) {
124             rlim_cur = value;
125             rlim_max = value;
126         }
127 
128         /**
129          * Soft limit.
130          */
131         public long rlim_cur = -1;  /* Soft limit */
132 
133         /**
134          * Hard limit (ceiling for rlim_cur).
135          */
136         public long rlim_max = -1;  /* Hard limit (ceiling for rlim_cur) */
137 
138         @Override
139         public String toString() {
140             return String.format( "rlim_cur: %,d, , rlim_max: %,d" , rlim_cur, rlim_max );
141         }
142         
143     }

145     public static RlimitStruct getrlimit( RLIMIT resource ) throws Exception {
146 
147         assertPlatform();
148 
149         RlimitStruct result = new RlimitStruct();
150 
151         if ( delegate.getrlimit( resource.value(), result ) != 0 ) {
152             throw new Exception( errno.strerror() );
153         }
154 
155         return result;
156         
157     }
158 
159     public static void setrlimit( RLIMIT resource, RlimitStruct limit ) throws Exception {
160 
161         assertPlatform();
162         
163         if ( delegate.setrlimit( resource.value(), limit ) != 0 ) {
164             throw new PlatformException( "Unable to setrlimit on: " + resource );
165         }
166 
167         // now make sure we actually wrote the correct resources.
168 
169         RlimitStruct after = getrlimit( resource );
170 
171         if ( after.rlim_cur != limit.rlim_cur ||
172              after.rlim_max != limit.rlim_max ) {
173 
174             throw new Exception( String.format( "Limit set incorrectly (requested %s vs %s) on " + resource ,
175                                                 limit, after ) );
176             
177         }
178         
179     }

181     private static void assertPlatform() {
182         
183         if ( Platform.isLinux() || Platform.isDarwin() ) {
184             return;
185         }
186             
187         throw new RuntimeException( "Platform is not supported: " + Platform.getOS() );
188
