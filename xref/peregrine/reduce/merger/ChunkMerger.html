36  /**
37   * http://en.wikipedia.org/wiki/External_sorting
38   * 
39   * <p>
40   * One example of external sorting is the external merge sort algorithm, which
41   * sorts chunks that each fit in RAM, then merges the sorted chunks
42   * together.[1][2] For example, for sorting 900 megabytes of data using only 100
43   * megabytes of RAM:
44   *
45   * <p>
46   * Read 100 MB of the data in main memory and sort by some conventional method,
47   * like quicksort.
48   * 
49   * <p>
50   * Write the sorted data to disk.
51   * 
52   * <p>
53   * Repeat steps 1 and 2 until all of the data is in sorted 100 MB chunks (there are
54   * 900MB / 100MB = 9 chunks), which now need to be merged into one single output
55   * file.
56   * 
57   * <p>
58   * Read the first 10 MB (= 100MB / (9 chunks + 1)) of each sorted chunk into input
59   * buffers in main memory and allocate the remaining 10 MB for an output
60   * buffer. (In practice, it might provide better performance to make the output
61   * buffer larger and the input buffers slightly smaller.)
62   * 
63   * <p>
64   * Perform a 9-way merge and store the result in the output buffer. If the output
65   * buffer is full, write it to the final sorted file, and empty it. If any of the 9
66   * input buffers gets empty, fill it with the next 10 MB of its associated 100 MB
67   * sorted chunk until no more data from the chunk is available. This is the key
68   * step that makes external merge sort work externally -- because the merge
69   * algorithm only makes one pass sequentially through each of the chunks, each
70   * chunk does not have to be loaded completely; rather, sequential parts of the
71   * chunk can be loaded as needed.
72   * 
73   * <p>
74   * http://en.wikipedia.org/wiki/Merge_algorithm
75   * 
76   * <p>
77   * Merge algorithms generally run in time proportional to the sum of the lengths of
78   * the lists; merge algorithms that operate on large numbers of lists at once will
79   * multiply the sum of the lengths of the lists by the time to figure out which of
80   * the pointers points to the lowest item, which can be accomplished with a
81   * heap-based priority queue in O(log n) time, for O(m log n) time, where n is the
82   * number of lists being merged and m is the sum of the lengths of the lists. When
83   * merging two lists of length m, there is a lower bound of 2m - 1 comparisons
84   * required in the worst case.
85   * 
86   * <p>
87   * The classic merge (the one used in merge sort) outputs the data item with the
88   * lowest key at each step; given some sorted lists, it produces a sorted list
89   * containing all the elements in any of the input lists, and it does so in time
90   * proportional to the sum of the lengths of the input lists.
91   * 
92   * 
93   */
94  public class ChunkMerger implements Closeable {
95  
96      private static final Logger log = Logger.getLogger();
97  
98      private SortListener listener = null;
99  
100     public int entries = 0;
101         
102     private Partition partition;
103 
104     private List<JobOutput> output;
105 
106     private Task task = null;
107 
108     private List<SequenceReader> input;
109 
110     private SortComparator sortComparator = null;
111     
112     public ChunkMerger() {
113     }
114 
115     public ChunkMerger( Task task,
116                         SortListener listener,
117                         Partition partition,
118                         List<SequenceReader> input,
119                         List<JobOutput> output,
120                         SortComparator sortComparator ) {
121 
122         this.task = task;
123         this.listener = listener;
124         this.partition = partition;
125         this.input = input;
126         this.output = output;
127         this.sortComparator = sortComparator;
128         
129     }
130 
131     public void merge() throws IOException {
132         merge( null );
133     }
134     
135     public void merge( ChunkWriter writer ) throws IOException {
136 
137         try {
138         
139             //TODO: if the input length is zero or one then we are done, however
140             //everything will need to be written to the writer first which is
141             //somewhat inefficient so we should try to reference the original file
142             //that was supplied and just return that directly.  In practice though
143             //this will only be a single 100MB file so this is not the end of the
144             //world.
145 
146             if ( input.size() == 0 ) {
147                 log.info( "No input to merge." );
148                 return;
149             }
150 
151             log.info( "Merging %,d readers." , input.size() );
152             
153             MergerPriorityQueue queue = new MergerPriorityQueue( input, sortComparator );
154                             
155             SortResult sortResult = new SortResult( writer, listener );
156 
157             while( true ) {
158                 
159                 MergeQueueEntry entry = queue.poll();
160 
161                 if ( entry == null )
162                     break;
163 
164                 task.assertActiveJob(); 
165 
166                 sortResult.accept( new SortEntry( entry.key, StructReaders.unsplice( entry.value ) ) );
167 
168                 ++entries;
169 
170             }
171 
172             sortResult.flush();
173 
174             if ( writer != null )
175                 writer.flush();
176             
177             sortResult.close();
178 
179             if ( writer != null )         
180                 writer.close();
181 
182             log.info( "Merged %,d entries for %s" , entries, partition );
183 
184             if( output != null )
185                 new Flusher( output ).flush();
186 
187         } catch ( Throwable t ) {
188             throw new IOException( "Unable to merge chunks: " + input, t );
189         }
190 
191     }
192 
193     public void close() throws IOException {
194 
195         //TODO: these should use one closer.
196         if ( output != null ) {
197             new Closer( output ).close();
198         }
199 
200         new Closer( input ).close();
201     }
202 
203 }
