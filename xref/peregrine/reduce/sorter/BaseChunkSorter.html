16  package peregrine.reduce.sorter;
17  
18  import peregrine.io.util.Closer;
19  import peregrine.sort.SortComparator;
20  
21  import java.io.IOException;
22  import java.util.ArrayList;
23  import java.util.List;
24  
25  /**
26   * Base sorter which includes the core logic behind sorting.
27   * 
28   */
29  public class BaseChunkSorter {
30  
31      private SortComparator comparator;
32  
33      public BaseChunkSorter( SortComparator comparator ) {
34          this.comparator = comparator;
35      }
36  
37      public KeyLookup sort( KeyLookup input ) 
38          throws IOException {
39  
40          // we are already sorted so just return.
41          if ( input.size() <= 1 ) {
42              return input;
43          }
44  
45          int middle = input.size() / 2; 
46  
47          KeyLookup left  = null;
48          KeyLookup right = null;
49  
50          try {
51  
52              left  = input.slice( 0, middle - 1 );
53              right = input.slice( middle, input.size() - 1 );
54  
55              left  = sort( left );
56              right = sort( right );
57  
58              // determine which merge structure to use... if this is the LAST merger
59              // just write the results to disk.  Writing to memory and THEN writing
60              // to disk would just waste CPU time.
61  
62              return merge( left, right );
63              
64          } finally {
65  
66              // close both the left and right inputs to return their memory.
67              new Closer( left, right, input ).close();
68  
69          }
70          
71      }
72  
73      private KeyLookup merge( KeyLookup left,
74                               KeyLookup right )
75          throws IOException {
76          
77          KeyLookup result = new KeyLookup( left.size() + right.size(), left.buffers );
78          
79          // now merge both left and right and we're done.
80          List<KeyLookup> list = new ArrayList<KeyLookup>( 2 );
81          list.add( left );
82          list.add( right );
83  
84          SorterPriorityQueue queue = new SorterPriorityQueue( list, comparator );
85  
86          while( true ) {
87              
88              SortQueueEntry poll = queue.poll();
89  
90              if ( poll == null )
91                  break;
92              
93              result.next();
94              result.set( poll.entry );
95              
96          }
97  
98          result.reset();
99          return result;
100         
101     }
