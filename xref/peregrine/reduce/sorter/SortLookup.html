23  import peregrine.io.chunk.CompositeChunkReader;
24  import peregrine.util.Integers;

27  import java.io.Closeable;
28  import java.io.IOException;

31  /**
32   * Allocate a lookup for N records and a separate direct buffer for storing
33   * these records.  We include a method to determine if a key, which is about to
34   * be written, would have too much capacity for writes.
35   */
36  public class SortLookup implements KeyLookup, Closeable {
37  
38      protected int endInclusive;
39  
40      protected int size;
41  
42      protected int index = -1;
43  
44      protected int startInclusive = -1;
45  
46      protected ChannelBuffer buffer;
47  
48      protected OffsetIndex offsetIndex;
49  
50      protected DirectMemoryAllocator allocator;
51  
52      // true when this is a temporary / intermediate buffer.
53      protected boolean temporary = false;
54  
55      private DirectMemoryRegion directMemoryRegion;
56  
57      private SortLookup() {}
58  
59      public SortLookup( DirectMemoryAllocator allocator, CompositeChunkReader reader)
60        throws IOException {
61  
62          this( allocator, reader.count(), computeDataCapacity( reader.count(), reader.length() ) );
63  
64          while ( reader.hasNext() ) {
65  
66              // advance the reader
67              reader.next();
68  
69              // advance the lookup
70              next();
71  
72              set( new Record( reader.key(), reader.value() ) );
73  
74          }
75  
76          reset();
77  
78      }
79  
80      public SortLookup( DirectMemoryAllocator allocator, int nrRecords, long capacity ) {
81  
82          this.allocator = allocator;
83  
84          // allocate 8 bytes per record for the offset so that we can read the
85          // key in the given index.   TODO. work on a better storage index for
86          // the lookup to get tighter usage of memory.
87  
88          this.offsetIndex = new OffsetIndex( allocator, nrRecords );
89  
90          if ( capacity > Integer.MAX_VALUE )
91              throw new IllegalArgumentException( "Required sort capacity too large: " + capacity );
92  
93          this.directMemoryRegion = allocator.allocate( (int) capacity );
94          this.buffer = directMemoryRegion.getBuffer();
95  
96          this.endInclusive = nrRecords - 1;
97          this.size = nrRecords;
98          this.startInclusive = 0;
99  
100     }
101 
102     @Override
103     public KeyLookup allocate(KeyLookup kl0, KeyLookup kl1 ) {
104 
105         SortLookup sl0 = (SortLookup)kl0;
106         SortLookup sl1 = (SortLookup)kl1;
107 
108         int nrRecords = kl0.size() + kl1.size();
109         long capacity = sl0.requiredCapacity() + sl1.requiredCapacity();
110 
111         SortLookup result = new SortLookup( allocator, nrRecords, capacity );
112         result.temporary = true;
113 
114         return result;
115 
116     }
117 
118     @Override
119     public boolean hasNext() {
120         return index < endInclusive;
121     }
122 
123     @Override
124     public void next() {
125         ++index;
126     }
127 
128     @Override
129     public int size() {
130         return size;
131     }
132 
133     @Override
134     public boolean isTemporary() {
135         return temporary;
136     }
137 
138     @Override
139     public void set(Record entry) {
140 
141         offsetIndex.set( index, buffer.writerIndex() );
142 
143         writeEntry( entry.getKey() );
144         writeEntry( entry.getValue() );
145 
146     }
147 
148     // write a struct reader and length prefix it.
149     private void writeEntry( StructReader sr ) {
150 
151         ChannelBuffer cb = sr.getChannelBuffer();
152 
153         int len = cb.writerIndex();
154 
155         buffer.writeInt( len );
156         buffer.writeBytes( cb );
157 
158     }
159 
160     @Override
161     public Record get() {
162 
163         int readerIndex = offsetIndex.get( index );
164 
165         if ( readerIndex < 0 )
166             throw new RuntimeException( String.format( "Offset %s invalid for index %s", readerIndex, index ) );
167 
168         buffer.readerIndex( readerIndex );
169 
170         StructReader key = readEntry();
171         StructReader value = readEntry();
172 
173         return new Record( key, value );
174 
175     }
176 
177     private StructReader readEntry() {
178 
179         int len = buffer.readInt();
180 
181         ChannelBuffer cb = buffer.readSlice( len );
182 
183         return new StructReader( cb );
184 
185     }
186 
187     @Override
188     public void reset() {
189         this.index = startInclusive - 1;
190     }
191 
192     /**
193      * @return The number of bytes required to store this in a new lookup.
194      * We use this to determine how much to store in a new allocate() call.
195      */
196     public long requiredCapacity() {
197 
198         int start = offsetIndex.get( startInclusive );
199 
200         int end = offsetIndex.get( endInclusive + 1 );
201 
202         if ( end == -1 ) {
203             end = buffer.capacity();
204         }
205 
206         return end - start;
207 
208     }
209 
210     // zero copy slice implementation.
211     @Override
212     public KeyLookup slice( int sliceStartInclusive, int sliceEndInclusive ) {
213 
214         SortLookup slice = new SortLookup();
215 
216         slice.offsetIndex = offsetIndex;
217         slice.buffer = buffer;
218 
219         slice.size  = (sliceEndInclusive - sliceStartInclusive) + 1;
220         slice.startInclusive = startInclusive + sliceStartInclusive;
221         slice.index  = slice.startInclusive - 1;
222         slice.endInclusive = slice.startInclusive + slice.size - 1;
223         slice.directMemoryRegion = directMemoryRegion;
224         slice.allocator = allocator;
225         slice.temporary = temporary;
226 
227         return slice;
228 
229     }
230 
231     @Override
232     public void close() throws IOException {
233 
234         offsetIndex.close();
235         directMemoryRegion.close();
236 
237     }
238 
239     public static long computeCapacity( int nrRecords, int dataLength ) {
240 
241         // this computes an UPPER bound for the capacity.  We could probably
242         // get MUCH better usage of memory but we need to compute the length of
243         // raw data ahead of time.
244         return computeOffsetIndexCapacity( nrRecords ) +
245                computeDataCapacity( nrRecords, dataLength );
246 
247     }
248 
249     // usage for the offset index
250     private static long computeOffsetIndexCapacity( int nrRecords ) {
251         return (nrRecords * (long)Integers.LENGTH);
252     }
253 
254     private static long computeDataCapacity( int nrRecords, int dataLength ) {
255 
256         return (nrRecords * (long)Integers.LENGTH ) + /* four bytes for each key    */
257                (nrRecords * (long)Integers.LENGTH ) + /* four bytes for each value  */
258                dataLength                             /* the data bytes             */
259             ;
260 
261     }
