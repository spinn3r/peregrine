18  import org.jboss.netty.buffer.ChannelBuffer;
19  import org.jboss.netty.buffer.ChannelBuffers;
20  import org.jboss.netty.handler.codec.http.QueryStringDecoder;
21  import org.jboss.netty.handler.codec.http.QueryStringEncoder;
22  import peregrine.util.StructMap;

24  import java.util.List;
25  import java.util.Map;

27  /**
28   * A message sent between two hosts.
29   */
30  public class Message extends StructMap implements MessageSerializable {
31  
32      public Message() { }
33  
34      public Message( String data ) {
35  
36          if ( data == null ) return;
37          
38          Map<String,List<String>> decoded
39              = new QueryStringDecoder( data ).getParameters();
40  
41          for( String key : decoded.keySet() ) {
42              put( key , decoded.get( key ).get(0) );
43          }
44  
45      }
46  
47      public Message( Object... args ) {
48          for( int i = 0; i < args.length; ++i ) {
49              put( args[i].toString(), args[i + 1].toString() );
50              ++i;
51          }
52      }
53      
54      public Message( Map map ) {
55          super( map );
56      }
57      
58      @Override
59      public String toString() {
60          
61          QueryStringEncoder encoder = new QueryStringEncoder( "" );
62  
63          for( String key : keys ) {
64  
65              Object value = delegate.get( key );
66  
67              if( value != null ) {
68                  encoder.addParam( key, value.toString() );
69              }
70              
71          }
72  
73          return encoder.toString();
74  
75      }
76  
77      /**
78       * Print a message which a human could read.
79       */
80      public String toDebugString() {
81  
82          StringBuilder buff = new StringBuilder();
83          
84          for( String key : keys ) {
85  
86              Object value = delegate.get( key );
87  
88              buff.append( String.format( "%15s = %s\n", key, value ) );
89              
90          }
91  
92          if ( buff.length() == 0 )
93              return "";
94          
95          return buff.toString().substring( 0, buff.length() - 1 ) ;
96  
97      }
98      
99      public Message getMessage( String key ) {
100         return new Message( getString( key ) );
101     }
102     
103     @Override
104     public Message toMessage() {
105         return this;
106     }
107 
108     @Override
109     public void fromMessage( Message message ) {
110         init( message.toMap() );
111     }
112 
113     public ChannelBuffer toChannelBuffer() {
114         return ChannelBuffers.wrappedBuffer( toString().getBytes() );
115     }
