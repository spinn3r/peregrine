16  package peregrine.shuffle;
17  
18  import org.jboss.netty.buffer.ChannelBuffer;
19  import peregrine.StructReader;
20  import peregrine.config.Config;
21  import peregrine.config.Partition;
22  import peregrine.os.MappedFileReader;
23  import peregrine.util.Hex;
24  
25  import java.io.Closeable;
26  import java.io.File;
27  import java.io.IOException;
28  import java.util.*;
29  
30  /**
31   * Read packets from a shuffle file for a given set of partitions (usually the
32   * primary).
33   */
34  public class ShuffleInputReader implements Closeable {
35  
36      /**
37       * The current packet index we're on.
38       */
39      private int packet_idx = 0;
40  
41      /**
42       * ALL known headers in this shuffle file.
43       */
44      protected List<ShuffleHeader> headers = new ArrayList<ShuffleHeader>();
45  
46      protected Map<Partition,ShuffleHeader> headersByPartition = new HashMap<Partition,ShuffleHeader>();
47      
48      protected ChannelBuffer buffer = null;
49  
50      /**
51       * The number of packet we need to read for all given partitions.
52       */
53      protected int nr_packets = 0;
54  
55      /**
56       * Used so that we can move over partitions.
57       * 
58       */
59      protected Iterator<Partition> partitionIterator = null;
60  
61      /**
62       * The current partition we're working on.
63       */
64      protected Partition current = null;
65      
66      protected MappedFileReader mappedFile = null;;
67  
68      protected Config config = null;
69      
70      public ShuffleInputReader( Config config, String path, final Partition partition ) throws IOException {
71          this( config, path, new ArrayList() {{ add( partition ); }} );
72      }
73          
74      public ShuffleInputReader( Config config, String path, List<Partition> partitions ) throws IOException {
75  
76          this.config = config;
77          
78          if ( config == null )
79              throw new NullPointerException( "config" );
80          
81          // pull out the header information 
82  
83          File file = new File( path );
84  
85          // NOTE: we read directly from a FileInputStream here because we need to
86          // use skip to jump over the preamble. This MAY be a slight performance
87          // issue but I doubt it.  There are only 12 bytes per partition and if
88          // the local partition has 25 partitions this is only 300 bytes and in
89          // the worse case scenario we read 288 extra bytes.  Ideally we would
90          // store the shuffle correctly so that the partitions this machine was
91          // responsible for reducing over were at the beginning of the shuffle
92          // data but in practice this may be a premature optimization.
93  
94          // mmap the WHOLE file. We won't actually use these pages if we don't
95          // read them so this make it less difficult to figure out what to map.
96  
97          this.mappedFile = new MappedFileReader( config, file );
98          
99          this.buffer = mappedFile.map();
100         
101         StructReader struct = new StructReader( buffer );
102 
103         struct.read( new byte[ ShuffleOutputWriter.MAGIC.length ] );
104 
105         int nr_partitions = struct.readInt();
106 
107         for ( int i = 0; i < nr_partitions; ++i ) {
108 
109             ShuffleHeader current = new ShuffleHeader();
110             
111             current.partition    = struct.readInt();
112             current.offset       = struct.readInt();
113             current.nr_packets   = struct.readInt();
114             current.count        = struct.readInt();
115             current.length       = struct.readInt();
116 
117             if ( current.partition   < 0 ||
118                  current.offset      < 0 ||
119                  current.nr_packets  < 0 ||
120                  current.count       < 0 ||
121                  current.length      < 0 ) {
122                 
123                 throw new IOException( "Header corrupted: " + current );
124                 
125             }
126 
127             // record this for usage later if necessary.
128             headers.add( current );
129             headersByPartition.put( new Partition( current.partition ), current );
130 
131         }
132 
133         // make sure we have headers for ALL the partitions we requested
134         for( Partition partition : partitions ) {
135 
136             ShuffleHeader header = headersByPartition.get( partition );
137             
138             if ( header == null ) {
139 
140                 throw new IOException( String.format( "Unable to find header for partition %s in file %s with headers %s",
141                                                       partition, path, headers ) );
142                 
143             }
144 
145             // determine the number of packets we need to read from all the partitions.
146             nr_packets += header.nr_packets;
147             
148         }
149         
150         partitionIterator = partitions.iterator();
151 
152         nextPartition();
153         
154     }
155 
156     private void nextPartition() {
157 
158         current = partitionIterator.next();
159 
160         ShuffleHeader header = headersByPartition.get( current );
161 
162         this.buffer.readerIndex( header.offset );
163         
164     }
165     
166     public ChannelBuffer getBuffer() {
167         return buffer;
168     }
169 
170     public ShuffleHeader getHeader( Partition partition ) {
171         return headersByPartition.get( partition );
172     }
173     
174     public boolean hasNext() throws IOException {
175         return packet_idx < nr_packets;
176     }
177     
178     public ShufflePacket next() throws IOException {
179 
180         if ( packet_idx >= nr_packets ) {
181             return null;
182         }
183 
184         ++packet_idx;
185 
186         int from_partition  = buffer.readInt();
187         int from_chunk      = buffer.readInt();
188         int to_partition    = buffer.readInt();
189         int len             = buffer.readInt();
190         int offset          = buffer.readerIndex();
191         
192         if ( to_partition != current.getId() ) {
193             // skip over this partition and move to the next one.
194             nextPartition();
195 
196             --packet_idx; /* this doesn't count */
197             return next();
198         }
199 
200         ChannelBuffer data = buffer.readSlice( len );
201         
202         // TODO: why is count -1 here?  That makes NO sense.
203         int count = -1;
204         
205         ShufflePacket pack = new ShufflePacket( from_partition, from_chunk, to_partition, offset, count, data );
206 
207         return pack;
208         
209     }
210 
211     @Override
212     public void close() throws IOException {
213 
214         mappedFile.close();
215 
216         // TODO fadvise away these pages now that we are done with them.
217         
218     }
219 
220     public static void main( String[] args ) throws IOException {
221 
222         String path = args[0];
223 
224         // debug code to dump a shuffle.
225 
226         List<Partition> partitions = new ArrayList();
227         
228         for( int i = 1; i < args.length; ++i ) {
229             partitions.add( new Partition( Integer.parseInt( args[i] ) ) );
230         }
231         
232         System.out.printf( "Reading from partitions: %s\n", partitions );
233 
234         ShuffleInputReader reader = new ShuffleInputReader( null, path, partitions );
235 
236         System.out.printf( "Headers: \n" );
237         for( ShuffleHeader header : reader.headers ) {
238             System.out.printf( "\t%s\n", header );
239         }
240         
241         while( reader.hasNext() ) {
242 
243             ShufflePacket pack = reader.next();
244 
245             System.out.printf( "pack: %s\n", pack );
246 
247             System.out.printf( "%s\n", Hex.pretty( pack.data ) );
248             
249         }
250 
251     }
