16  package peregrine.shuffle.receiver;
17  
18  import com.spinn3r.log5j.Logger;
19  import peregrine.config.Config;
20  import peregrine.io.util.Files;
21  
22  import java.io.IOException;
23  import java.util.concurrent.ConcurrentHashMap;
24  
25  /**
26   * Handles accepting shuffle data and rolling over shuffle files when buffers
27   * are full.
28   */
29  public class ShuffleReceiverFactory {
30  
31      private static final Logger log = Logger.getLogger();
32      
33      private ConcurrentHashMap<String,ShuffleReceiver> instances
34          = new ConcurrentHashMap<String,ShuffleReceiver>();
35  
36      protected Config config;
37  
38      public long lastFlushed = -1;
39      
40      public ShuffleReceiverFactory( Config config ) {
41          this.config = config;
42      }
43      
44      public ShuffleReceiver getInstance( String name ) {
45  
46          ShuffleReceiver result = instances.get( name );
47  
48          // double check idiom.  this doesn't happen very often so this should be
49          // fine and won't impact performance.
50          if ( result == null ) {
51  
52              synchronized( instances ) {
53  
54                  result = instances.get( name );
55  
56                  if ( result == null ) {
57  
58                      result = new ShuffleReceiver( config, name );
59                      instances.putIfAbsent( name, result );
60                      result = instances.get( name );
61                      
62                  } 
63  
64              }
65              
66          }
67  
68          return result;
69          
70      }
71  
72      /**
73       * Close all shufflers and flush their output to disk.
74       */
75      public void flush() throws IOException {
76  
77          if ( instances.size() == 0 )
78              return;
79          
80          log.info( "Flushing %,d shufflers...", instances.size() );
81  
82          for( ShuffleReceiver current : instances.values() ) {
83              current.close();
84          }
85  
86          lastFlushed = System.currentTimeMillis();
87  
88          log.info( "Flushing %,d shufflers...done", instances.size() );
89  
90          // now throw the current instances away because we can't use them any
91          // more and this will also free up memory.
92          instances = new ConcurrentHashMap();
93              
94      }
95  
96      public void purge( String name ) throws IOException {
97  
98          // only done so that we can benchmark the performance of certain
99          // algorithms.
100         
101         if ( config.getPurgeShuffleData() == false )
102             return;
103         
104         String dir = config.getShuffleDir( name );
105         
106         log.info( "Purging shuffler directory: %s", dir );
107 
108         Files.remove( dir );
109         
110     }
111     
112     public long lastFlushed() {
113         return lastFlushed;
114     }
115     
116 }
