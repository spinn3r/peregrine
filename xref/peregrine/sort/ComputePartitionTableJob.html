30  /**
31   * Map reduce job which computes the partition routing table for sorting data.
32   */
33  public class ComputePartitionTableJob {
34  
35      private static final Logger log = Logger.getLogger();
36  
37      /**
38       * The maximum percentage to tolerate in key distribution.  If we have a
39       * percentage gap greater than this we abort.
40       */
41      public static int MAX_DISTRIBUTION_PERC_DELTA = 10;
42      
43      /**
44       * The maximum number of keys to read into memory to compute the training
45       * set..
46       * 
47       * <p> TODO: Technically this should be a function of the number of
48       * partitions and data because our accuracy will fail with the total number of
49       * partitions we have.
50       * TODO: this should ALSO be kept outside of the JVM and allocated from an
51       * external buffer.  We could/should probably use something like the sort
52       * buffer size which isn't going to be used during reduce jobs.
53       */
54      public static int MAX_TRAIN_SIZE = 100000;
55  
56      /**
57       * The point at which we essentially consider that we have no data and don't
58       * even bother computing the partition table.
59       */
60      public static int NO_DATA_THRESHOLD = 1000;
61      
62      /**
63       * <p>
64       * The key length for partition boundaries.  Eight (8) bites for the first
65       * component and 8 bytes for the second component.
66       *
67       * <p>
68       * The first component is the value we are sorting by.  Eight bytes give us
69       * enough room for doubles/longs.
70       *
71       * <p>
72       * The second component is an 8 byte value for the hashcode for that item.
73       * This way we have enough bytes to split data even when the sorting column
74       * is the same for 2^64 items.
75       * 
76       * <p>In the future we should consider a way to have custom width sort keys.
77       */
78      public static final int KEY_LEN = 16;
79  
80      public static final StructReader LAST_BOUNDARY  = createByteArray( 127, KEY_LEN );
81      
82      /**
83       * Create a byte array of the given values.
84       */
85      private static StructReader createByteArray( int val, int len ) {
86  
87          byte[] data = new byte[ len ];
88  
89          for( int i = 0; i < data.length; ++i ) {
90              data[i] = (byte)val;
91          }
92  
93          return StructReaders.wrap( data );
94          
95      }

97      public static List<StructReader> computePartitionBoundaries( Config config,
98                                                                   Partition part,
99                                                                   SortComparator comparator,
100                                                                  List<StructReader> train ) {
101 
102         log.info( "Training set size is: %,d items", train.size() );
103 
104         List<StructReader> result = new ArrayList<StructReader>();
105 
106         if ( train.size() <= NO_DATA_THRESHOLD ) {
107 
108             // It is totally reasonable to want to sort a file with no
109             // entries.
110             
111             log.warn( "No training data for job." );
112 
113             return result;
114             
115         }
116 
117         Collections.sort( train, comparator );
118 
119         // now down sample the training set so that we can grok what is
120         // happening by looking at a smaller number of items.  In practice the
121         // partition boundaries won't be off significantly using this technique.
122 
123         for( StructReader current : summarize( train, 80 ) ) {
124             log.info( "Training data set summary on %s %s", part, format( current ) );
125         }
126         
127         int nr_partitions = config.getMembership().size();
128 
129         result = summarize( train, nr_partitions );
130 
131         // we have to remove the last partition as it is pointless essentially.
132 
133         result.remove( result.size() - 1 );
134 
135         // add the max partition boundary.
136         result.add( LAST_BOUNDARY );
137 
138         for( StructReader current : result ) {
139             log.info( "Resulting partition table %s %s", part, format( current ) );
140         }
141 
142         return result;
143 
144     }

150     /**
151      * Summarize the given list by taking a reading every list.size() / count
152      * items and return the result.  
153      */
154     public static List<StructReader> summarize( List<StructReader> list, int count ) {
155 
156         int width = list.size() / count;
157 
158         int offset = 0;
159 
160         List<StructReader> result = new ArrayList();
161         
162         for( int i = 0; i < count; ++i ) {
163 
164             offset += width;
165             
166             result.add( list.get( offset - 1) );
167             
168         }
169 
170         return result;
171         
172     }

187     public static class Map extends Mapper {
188 
189         private static final Logger log = Logger.getLogger();
190 
191         /**
192          * The training data.  We sort this list and then take trains at
193          * partition boundaries.
194          */
195         private List<StructReader> train = new ArrayList<StructReader>( MAX_TRAIN_SIZE );
196 
197         /**
198          * The testing data for verifying the training data.
199          */
200         private List<StructReader> test  = new ArrayList<StructReader>( MAX_TRAIN_SIZE );
201 
202         private SortComparator comparator = null;
203 
204         private Random rand = new Random();
205         
206         /**
207          * The ID of the current record we are processing.
208          */
209         private int id = 0;
210         
211         @Override
212         public void init( Job job, List<JobOutput> output ) {
213 
214             super.init( job, output );
215             comparator = getSortComparator( job );
216 
217         }
218 
219         @Override
220         public void map( StructReader key,
221                          StructReader value ) {
222 
223             // NOTE: we have to call toByteArray here because we are keeping
224             // these values around for a long time and the maps might not be
225             // valid when we try to read them later.
226 
227             // NOTE: We must use a random key here because we don't have a full
228             // range of key data in this sample since we only read one chunk to
229             // train and the keys are sorted.  If we didn't use a random value
230             // we would have just a subset of the keys and our distribution
231             // wouldn't be perfect for data sets with a non-linear distribution
232             // of values (IE where everything is zero and key value range
233             // becomes important).
234 
235             // FIXME: this breaks our ability to sort variable length keys.
236 
237             key   = StructReaders.wrap( rand.nextLong() );
238             value = StructReaders.wrap( value.toByteArray() );
239 
240             if ( ++id % 2 == 0 ) {
241                 addTrain( key, value );
242             } else {
243                 addTest( key, value );
244             }
245 
246         }
247 
248         /**
249          * Add a data point to the training set. 
250          */
251         protected void addTrain( StructReader key, StructReader value ) {
252             addPoint( key, value, train );
253         }
254 
255         /**
256          * Add a data point to the testing set. 
257          */
258         protected void addTest( StructReader key, StructReader value ) {
259             addPoint( key, value, test );
260         }
261 
262         /**
263          * Add a sort key data point to the given list.
264          */
265         protected void addPoint( StructReader key,
266                                  StructReader value,
267                                  List<StructReader> list ) {
268 
269             // we are done sampling.
270             if ( list.size() > MAX_TRAIN_SIZE )
271                 return;
272 
273             list.add( comparator.getSortKey( key , value ) );
274 
275         }
276 
277         @Override
278         public void close() throws IOException {
279 
280             long nr_partitions = config.getMembership().size();
281 
282             Collections.sort( train, comparator );
283 
284             if ( train.size() <= NO_DATA_THRESHOLD  ) {
285 
286                 log.warn( "Insufficient number of values to sort: %s", train.size() );
287 
288                 // write all data to the last boundary position.  It doesn't
289                 // matter where we write these since there are no records being
290                 // written.
291                 for( long i = 0; i < nr_partitions; ++i ) {
292                     StructReader key = StructReaders.wrap( i );
293                     emit( key, LAST_BOUNDARY );
294                 }
295 
296                 return;
297 
298             }
299 
300             // If we have too few training data points we have to create
301             // synthetic data and then generate new training data and then use
302             // the remaining eight(8) bytes to distribute values among the
303             // cluster.  The problem is that since we only read one chunk we're
304             // not getting the full key space but we can just create synthetic
305             // keys anyway.  The advantage to this approach is that we can
306             // re-use a lot of existing code.
307 
308             // ALWAYS making synthetic partitions when we have less than
309             // MAX_TRAIN_SIZE ... this train size should be our target so that
310             // we have fined grained partitioning of the data.
311             
312             if ( train.size() < MAX_TRAIN_SIZE ) {
313                 log.warn( "Training data set it small so distribution may be slightly skewed: %s", train.size() );
314             }
315 
316             log.info( "Working with %,d training set entries. " , train.size() );
317 
318             //at this point we should have all the trains, sort then and then
319             //determine partition boundaries.
320 
321             List<StructReader> boundaries = computePartitionBoundaries( config, getPartition(), comparator, train );
322 
323             //Test if we have >= MAX_TRAIN_SIZE records in the test data as
324             //there isn't enough values for a skew to even matter and our test()
325             //algorithm will probably be incorrect.
326             
327             if ( test.size() >= MAX_TRAIN_SIZE ) {
328                 test( boundaries );
329             } else {
330                 log.warn( "Testing data size is small. Skipping test since it would be inaccurate." );
331             }
332             
333             long partition_id = 0;
334 
335             for( StructReader boundary : boundaries ) {
336                 StructReader key = StructReaders.wrap( partition_id );
337                 emit( key, boundary );
338                 ++partition_id;
339             }
340             
341         }

343         private void test( List<StructReader> boundaries ) {
344 
345             GlobalSortPartitioner partitioner = new GlobalSortPartitioner();
346             partitioner.init( job, boundaries );
347 
348             int nr_partitions = config.getMembership().size();
349 
350             TreeMap<Partition, Integer> hits = new TreeMap<Partition, Integer>();
351             
352             for( int i = 0; i < nr_partitions; ++i ) {
353                 hits.put( new Partition( i ), 0 );
354             }
355 
356             // now partition around all the testing points.
357             for( StructReader sr : test ) {
358 
359                 Partition part = partitioner.partition( sr );
360                 hits.put( part, hits.get( part ) + 1 );
361                 
362             }
363 
364             //analyze the hits and throw an Exception if they don't look acceptable.
365             assertDistribution( toPerc( hits ) );
366 
367             log.info( "Test results for %,d testing points and %,d training points: %s",
368                       test.size(), train.size(), hits );
369             
370         }
371 
372         private void assertDistribution( java.util.Map<Partition, Integer> perc ) {
373 
374             int last = -1;
375 
376             for( int p : perc.values() ) {
377 
378                 if ( last > -1 && Math.abs( last - p ) > MAX_DISTRIBUTION_PERC_DELTA  ) {
379                     throw new RuntimeException( "Key distribution is skewed: " + perc );
380                 }
381 
382                 last = p;
383                 
384             }
385             
386         }
387 
388         private java.util.Map<Partition, Integer> toPerc( java.util.Map<Partition, Integer> data ) {
389 
390             int sum = 0;
391 
392             for( int val : data.values() ) {
393                 sum += val;
394             }
395 
396             java.util.Map<Partition, Integer> result = new HashMap();
397 
398             for( Partition part : data.keySet() ) {
399                 int perc = Integers.perc( data.get( part ) , sum );
400                 result.put( part , perc );
401             }
402 
403             return result;
404             
405         }

417     public static class Reduce extends Reducer {
418 
419         private static final Logger log = Logger.getLogger();
420 
421         private List<StructReader> boundaries = new ArrayList();
422 
423         private SortComparator comparator = null;
424         
425         @Override
426         public void init( Job job, List<JobOutput> output ) {
427 
428             super.init( job, output );
429             comparator = getSortComparator( job );
430 
431         }
432 
433         @Override
434         public void reduce( StructReader key, List<StructReader> values ) {
435 
436             StructReader value = mean( values );
437             //StructReader value = median( values );
438 
439             log.info( "Going to use final broadcast partition boundary: %s", Hex.encode( value ) );
440 
441             boundaries.add( value );
442 
443             if ( boundaries.size() == getConfig().getMembership().size() ) {
444                 emit( key, StructReaders.splice( boundaries ) );
445             }
446             
447         }
448 
449         protected StructReader mean( List<StructReader> list ) {
450 
451             BigInteger sum = BigInteger.valueOf( 0 );
452 
453             for( StructReader current : list ) {
454                 BigInteger bigInteger = new BigInteger( current.toByteArray() );
455                 sum = sum.add( bigInteger );
456             }
457 
458             BigInteger mean = sum.divide( BigInteger.valueOf( list.size() ) );
459 
460             StructReader median = median( list );
461 
462             // make sure the mean is the same byte width as the median.
463             byte[] padded = new byte[ median.length() ];
464             byte[] data = mean.toByteArray();
465 
466             System.arraycopy( data, 0, padded, padded.length - data.length, data.length );
467 
468             return StructReaders.wrap( padded );
469 
470         }
471 
472         private StructReader median( List<StructReader> values ) {
473             Collections.sort( values, comparator );
474             return values.get( values.size() / 2 );
475         }
