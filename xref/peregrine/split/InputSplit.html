16  package peregrine.split;
17  
18  import org.jboss.netty.buffer.ChannelBuffer;
19  import org.jboss.netty.buffer.ChannelBuffers;
20  import peregrine.io.util.Closer;
21  
22  import java.io.Closeable;
23  import java.io.File;
24  import java.io.FileInputStream;
25  import java.io.IOException;
26  import java.nio.ByteBuffer;
27  import java.nio.channels.FileChannel;
28  
29  /**
30   * A region of a file which has a deterministic size and ends on a record
31   * boundary.  This way we can parse the file easily.  We can also still use a
32   * ChannelBuffer/ByteBuffer as a backing since we can make it each one less than
33   * 2GB but aggregate them on files <b>larger</b> than 2GB.
34   */
35  public class InputSplit implements Closeable {
36  
37      private long start = 0;
38  
39      private long end = 0;
40  
41      private File file;
42  
43      private FileInputStream fis = null;
44  
45      private FileChannel channel = null;
46  
47      public InputSplit( long start,
48                         long end,
49                         File file ) {
50          this.start = start;
51          this.end = end;
52          this.file = file;
53      }
54  
55      public ChannelBuffer getChannelBuffer() throws IOException {
56  
57          fis = new FileInputStream( file );
58          channel = fis.getChannel();
59  
60          long length = end - start;
61  
62          ByteBuffer buff = channel.map( FileChannel.MapMode.READ_ONLY, start , length );
63  
64          return ChannelBuffers.wrappedBuffer( buff );
65  
66      }
67      
68      public String toString() {
69          return String.format( "%s, start=%,d , end=%,d" , file.getPath(), start, end );
70      }
71  
72      @Override
73      public void close() throws IOException {
74          new Closer( fis, channel );
75      }
76  
77  }
