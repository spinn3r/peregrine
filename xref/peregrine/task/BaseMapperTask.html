40  /**
41   * Base task for all task that read input from external systems or the 
42   * filesystem (or pipes).  In practice this boils down to map and merge tasks.
43   */
44  public abstract class BaseMapperTask extends BaseTask implements Callable {
45  
46      private static final Logger log = Logger.getLogger();
47  
48      /**
49       * This tasks partition listeners.
50       */
51      protected List<ChunkStreamListener> listeners = new ArrayList();
52  
53      protected List<SequenceReader> jobInput = null;
54  
55      protected MapperChunkStreamListener mapperChunkStreamListener = null;
56      
57      /**
58       * Run startup just on Mapper and Merger tasks.
59       */
60      public void init( Config config, Work work, Class delegate ) throws IOException {
61  
62          mapperChunkStreamListener = new MapperChunkStreamListener();
63          
64          super.init( config, work, delegate );
65  
66          // make all shuffle dirs for the shuffle output paths to make sure we
67          // have an empty set input for the shuffle data as opposed to missing
68          // data which means we tried to read from something that didn't exist
69          for( OutputReference ref : output.getReferences() ) {
70  
71              if ( ref instanceof ShuffleOutputReference ) {
72  
73                  ShuffleOutputReference shuffleOutput = (ShuffleOutputReference)ref;
74  
75                  String shuffle_dir = config.getShuffleDir( shuffleOutput.getName() );
76  
77                  File dir = new File( shuffle_dir );
78  
79                  if ( ! dir.exists() ) {
80                      Files.mkdirs( dir );
81                  }
82                  
83              }
84  
85          }
86  
87      }
88  
89      /**
90       * Construct a set of ChunkReaders (one per input source) for the given
91       * input.
92       */
93      protected List<SequenceReader> getJobInput() throws IOException {
94  
95          listeners.add( mapperChunkStreamListener );
96  
97          for( ShuffleJobOutput current : shuffleJobOutput ) {
98              listeners.add( current );
99          }
100 
101         List<SequenceReader> readers = new ArrayList();
102 
103         for( int i = 0; i < getInput().getReferences().size(); ++i ) {
104 
105         	InputReference inputReference  = getInput().getReferences().get( i );
106         	WorkReference  workReference   = getWork().getReferences().get( i );
107         	
108             if ( inputReference instanceof BroadcastInputReference ) {
109             	// right now we handle broadcast input differently.
110                 continue;
111             }
112 
113             IODriver driver = IODriverRegistry.getInstance( inputReference.getScheme() );
114             
115             // see if it is registered as a driver.
116             if ( driver != null ) {
117 
118                 JobInput ji = driver.getJobInput( config, job, inputReference, workReference );
119                 ji.addListeners( listeners );
120                 
121                 readers.add( ji );
122                 continue;
123             }
124 
125             throw new IOException( "Reference not supported: " + inputReference );
126             
127         }
128 
129         return readers;
130         
131     }
132 
133     protected String getPointer() {
134 
135         List<String> pointers = new ArrayList();
136 
137         for( SequenceReader reader : jobInput ) {
138 
139             if ( reader instanceof LocalPartitionReader )
140                 pointers.add( reader.toString() );
141             
142         }
143 
144         return Strings.join( pointers, "," );
145         
146     }
147 
148     protected String getNonce() {
149 
150         // host, job start time, and the pointer we are on
151         return Base64.encode( SHA1.encode( String.format( "%s:%s:%s", config.getHost(), started, getPointer() ) ) );
152         
153     }
154 
155     /**
156      * 
157      * 
158      * Make sure to always flush the output between chunks. This is only 1 flush
159      * per every 100MB or so and isn't the end of the world.
160      * 
161      */
162     class MapperChunkStreamListener implements ChunkStreamListener {
163 
164         public int lastChunk = -1;
165 
166         //private BaseMapper baseMapper = (BaseMapper)jobDelegate;
167         
168         public void onChunk( ChunkReference ref ) {
169 
170     	    log.info( "Handling chunk: %s for %s" , ref, getClass().getName() );
171 
172             ++lastChunk;
173             
174             // get the first nonce... 
175             if ( nonce == null ) {
176                 nonce = getNonce();
177             }
178 
179         }
180         
181         public void onChunkEnd( ChunkReference ref ) {
182 
183             try {
184 
185                 // fire onChunkEnd on the mapper so that intermediate chunk data
186                 // can be sent to broadcast shuffles.
187                 // baseMapper.onChunkEnd();
188                 
189                 // first try to flush job output.
190                 new Flusher( getJobOutput() ).flush();
191 
192                 pointer = getPointer();
193                 
194                 // update the nonce now ... 
195                 nonce = getNonce();
196                 
197             } catch ( IOException e ) {
198                 throw new RuntimeException( e );
199             }
200 
201         }
202 
203     }
