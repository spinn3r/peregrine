23  import peregrine.io.JobOutput;

25  import java.io.Closeable;
26  import java.io.IOException;
27  import java.util.List;

29  /**
30   * Represents a basic Map/Merge/Reduce code backend provided by a developer for
31   * running their job.  emit, setup, teardown, etc.
32   * 
33   */
34  public interface JobDelegate extends Closeable {
35  
36      public void setBroadcastInput( List<BroadcastInput> broadcastInput );
37  	
38      /**
39       * Init this job with the output it is supposed to handle.
40       */
41      public void init( Job job, List<JobOutput> output );
42      
43      /**
44       * Emit a key / value pair from the job.
45       */
46      public void emit( StructReader key, StructReader value );
47      
48      /**
49       * Cleanup after this job.  Close all output, etc.
50       */
51      public void close() throws IOException;
52  
53      /**
54       * Get the partition that a job is executing over.  This is mostly used for
55       * reference purposes for a job to perform potentially partition specific
56       * operations and for unit testing.
57       */
58      public Partition getPartition();
59  
60      public void setPartition( Partition partition );
61  
62      public Config getConfig();
63  
64      public void setConfig( Config config );
65  
66  }
