17  package peregrine.worker.clientd.requests;
18  
19  import peregrine.StructReader;
20  import peregrine.sort.StrictStructReaderComparator;
21  
22  /**
23   *
24   * A request for a set of keys or a scan request against the backend database from
25   * the client.  This is provided as it allows us to keep track of whether request
26   * was sent to the client (complete) and to get metadata about the client request.
27   */
28  public abstract class BackendRequest implements Comparable<BackendRequest>, RequestSizeable {
29  
30      // a strict comparator must be used or we are going to miss keys on disk.
31      private static final StrictStructReaderComparator comparator = new StrictStructReaderComparator();
32  
33      private ClientBackendRequest client = null;
34  
35      // a sibling request with the same key.  Used by the BackendRequestIndex.
36      private boolean hasSibling = false;
37  
38      // true if this request has completed and was either found or not found.
39      private boolean complete = false;
40  
41      private boolean found = false;
42  
43      private StructReader seekKey;
44  
45      protected BackendRequest(ClientBackendRequest client) {
46          this.client = client;
47      }
48  
49      protected BackendRequest(ClientBackendRequest client, StructReader seekKey ) {
50          this.client = client;
51          this.seekKey = seekKey;
52      }
53  
54      public ClientBackendRequest getClient() {
55          return client;
56      }
57  
58      /**
59       * True if we have completed looking for this request.  A request can be
60       * complete regardless of whether we have found it or not.
61       */
62      public boolean isComplete() {
63          return complete;
64      }
65  
66      public void setComplete(boolean complete) {
67          this.complete = complete;
68      }
69  
70      /**
71       * True if we have found the key we are looking for.
72       */
73      public boolean isFound() {
74          return found;
75      }
76  
77      public void setFound(boolean found) {
78          this.found = found;
79      }
80  
81      /**
82       *  Visit the given key/value pair and emit it if we return true.  We also
83       *  may need to set this request to complete if this serves the request.
84       *  For reading and individual key this is complete if the key request
85       *  matches the key we are serving.  For scan requests we have to keep
86       *  scanning until we hit the limit.
87       *
88       *  @return 0 if we're on the key we want (seek key), 1 if the key is
89       *  GREATER than our seek key, and -1 if it's less than our seek key.
90       */
91      public abstract void visit(StructReader key, StructReader value);
92  
93      /**
94       * The seekKey represents the entry that we're looking for with this
95       * backend request.  For GET requests it's just the key for that entry.
96       * For SCAN it's a bit different because we need to find that entry and
97       * then change the seekKey if we suspend the request and then resume.
98       */
99      public StructReader getSeekKey() {
100         return seekKey;
101     }
102 
103     public void setSeekKey(StructReader seekKey) {
104         this.seekKey = seekKey;
105     }
106 
107     public boolean hasSibling() {
108         return hasSibling;
109     }
110 
111     @Override
112     public int compareTo(BackendRequest backendRequest) {
113 
114         int diff = comparator.compare(getSeekKey(), backendRequest.getSeekKey() );
115 
116         // if there is no difference these guys are siblings.  Also, I don't like
117         // this technique purely from a code quality perspective because it relies
118         // on side-effect to mutate the object but the performance is better.
119         if ( diff == 0 ) {
120             hasSibling=true;
121             backendRequest.hasSibling=true;
122         }
123 
124         return diff;
125 
126     }
127 
128 }
